48 
49  struct sockaddr_storage sockaddr;
50  socklen_t len = sizeof(sockaddr);
51 
52  hSocket = accept(*listenSocket, (struct sockaddr*)&sockaddr, &len);
53 
54  // some security checks that the socket is really useable
55  if ((hSocket != INVALID_SOCKET) && (!csAddress.SetSockAddr((const struct sockaddr*)&sockaddr)))
56  LOG_ERROR("Warning: Unknown socket family", "NET");
57 
58  if (hSocket == INVALID_SOCKET)
59  {
60  int nErr = WSAGetLastError();
61  if (nErr != WSAEWOULDBLOCK)
62  LOG_ERROR("socket error accept failed - " + NetworkErrorString(nErr), "NET-PEERS");
63  markDestroy();
64  return;
65  }
66 
67  if (!IsSelectableSocket(hSocket))
68  {
69  LOG_ERROR("connection dropped: non-selectable socket - " + csAddress.toString(), "NET-PEERS");
70  markDestroy();
71  return;
72  }
73 
74  // According to the internet TCP_NODELAY is not carried into accepted sockets
75  // on all platforms. Set it again here just to be sure.
76  SetSocketNoDelay(hSocket);
77 
78  // get the semaphore grant without blocking
79  semaphoreGrant = cSemaphoreGrant(*semaphore, true);
80  if (!semaphoreGrant)
81  {
82  LOG("too many open connections - connection dropped", "NET-PEERS");
83  markDestroy();
84  return;
85  }
86 
87  LOG("connection accepted - " + csAddress.toString(), "NET-PEERS");
88 
89  // everything went smoothly, flag as connected
90  m_bConnected = true;
91 }

177 bool netPeers::tryConnectOutbound()
178 {
179  // this function will only be called when the connection is not yet established. we will also return true if it's already established
180  if (m_bConnected)
181  return true;
182 
183  // increment the connection try and update the timestamp
184  m_usConnectionTries++;
185  m_timeLastTry = GetTime();
186 
187  // when we're over a certain limit of connection tries (look at define.h for standard value), we don't even try it anymore to not stress the node
188  if (tooManyTriesOutbound())
189  return false;
190 
191  // alright we have the authorization to try to connect.
192  struct sockaddr_storage sockaddr;
193  socklen_t len = sizeof(sockaddr);
194  if (!csAddress.GetSockAddr((struct sockaddr*)&sockaddr, &len))
195  {
196  LOG_ERROR("Cannot connect: unsupported networkd - " + csAddress.toString(), "NET-PEERS");
197  return false;
198  }
199 
200  hSocket = socket(((struct sockaddr*)&sockaddr)->sa_family, SOCK_STREAM, IPPROTO_TCP);
201  if(hSocket == INVALID_SOCKET)
202  return false;
203 
204 #ifdef SO_NOSIGPIPE
205  int set = 1;
206  // Different way of disabling SIGPIPE on BSD
207  setsockopt(hSocket, SOL_SOCKET, SO_NOSIGPIPE, (void*)&set, sizeof(int));
208 #endif
209 
210  // Disable Nagle's algorithm
211  SetSocketNoDelay(hSocket);
212 
213  // Set to non-blocking
214  if (!SetSocketNonBlocking(hSocket, true))
215  {
216  CloseSocket(hSocket);
217  LOG_ERROR("Setting listening socket to non-blocking failed - " + NetworkErrorString(WSAGetLastError()), "NET-PEERS");
218  return false;
219  }
220 
221  if (connect(hSocket, (struct sockaddr*)&sockaddr, len) == SOCKET_ERROR)
222  {
223  int nErr = WSAGetLastError();
224  // WSAEINVAL is here because some legacy version of winsock uses it
225  if (nErr == WSAEINPROGRESS || nErr == WSAEWOULDBLOCK || nErr == WSAEINVAL)
226  {
227  struct timeval timeout = MillisToTimeval(MetaChain::getInstance().getNetworkManager()->getTimeout());
228  fd_set fdset;
229  FD_ZERO(&fdset);
230  FD_SET(hSocket, &fdset);
231  int nRet = select(hSocket + 1, NULL, &fdset, NULL, &timeout);
232  if (nRet == 0)
233  {
234  LOG_ERROR( "connection timed out - " + csAddress.toString(), "NET-PEERS" );
235  CloseSocket(hSocket);
236  return false;
237  }
238  if (nRet == SOCKET_ERROR)
239  {
240  LOG_ERROR("select() failed - " + csAddress.toString() + " - " + NetworkErrorString(WSAGetLastError()), "NET-PEERS" );
241  CloseSocket(hSocket);
242  return false;
243  }
244  socklen_t nRetSize = sizeof(nRet);
245 #ifdef _WIN32
246  if (getsockopt(hSocket, SOL_SOCKET, SO_ERROR, (char*)(&nRet), &nRetSize) == SOCKET_ERROR)
247 #else
248  if (getsockopt(hSocket, SOL_SOCKET, SO_ERROR, &nRet, &nRetSize) == SOCKET_ERROR)
249 #endif
250  {
251  LOG_ERROR("getsockopt() failed - " + csAddress.toString() + " - " + NetworkErrorString(WSAGetLastError()), "NET-PEERS" );
252  CloseSocket(hSocket);
253  return false;
254  }
255  if (nRet != 0)
256  {
257  LOG_ERROR("connect() failed after select() - " + csAddress.toString() + " - " + NetworkErrorString(nRet), "NET-PEERS" );
258  CloseSocket(hSocket);
259  return false;
260  }
261  }
262 #ifdef _WIN32
263  else if (WSAGetLastError() != WSAEISCONN)
264 #else
265  else
266 #endif
267  {
268  LOG_ERROR("connect() failed to - " + csAddress.toString() + " - " + NetworkErrorString(WSAGetLastError()), "NET-PEERS");
269  CloseSocket(hSocket);
270  return false;
271  }
272  }
273 
274  // find out if it's an selectable socket (only non windows
275  if (!IsSelectableSocket(hSocket))
276  {
277  LOG_ERROR("Cannot create connection: non-selectable socket created (fd >= FD_SETSIZE ?)", "NET-PEERS");
278  CloseSocket(hSocket);
279  return false;
280  }
281 
282 
283  LOG("Outgoing connection success - " + csAddress.toString(), "NET-PEERS");
284 
285  // everything is fine, yay!
286  m_bConnected = true;
287  return true;
288 }

290 bool netPeers::ReceiveMsgBytes(const char *pch, unsigned int nBytes, bool& complete)
291 {
292  complete = false;
293  int64_t nTimeMicros = GetTimeMicros();
294  LOCK(m_pcsvRecv);
295 
296  while (nBytes > 0)
297  {
298  // if we have more than MAX_MSG_QUEUE amount of messages to process, we skip reading. this prevents flodding from malicious nodes
299  if (m_queueMessages.size() >= MAX_MSG_QUEUE)
300  {
301  complete = true; // make sure the msg handler will be called to process those queued up msgs
302  return true;
303  }
304 
305  // read network data
306  int handled = m_netMsg.readData(pch, nBytes);
307 
308  if (handled < 0)
309  return false;
310 
311  pch += handled;
312  nBytes -= handled;
313 
314  if (m_netMsg.complete())
315  {
316  // update some time stats and set the message to be complete
317  m_netMsg.i64tTimeStart = nTimeMicros;
318  m_netMsg.i64tTimeDelta = GetTimeMicros() - nTimeMicros;
319  complete = true;
320 
321  // push the message into our queue
322  {
323  LOCK(pcsvQueue);
324  m_queueMessages.push(m_netMsg);
325  }
326 
327  // reset the pointer to a new Message
328  m_netMsg = netMessage();
329  }
330  }
331 
332  return true;
333 }
