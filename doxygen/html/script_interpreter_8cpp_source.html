51  bool scriptInterpreter::evalScript(SCRIPT_VERIFY flags, scriptError* serror)
52  {
53  static const CScriptNum bnZero(0);
54  static const CScriptNum bnOne(1);
55  static const valtype vchFalse(0);
56  static const valtype vchTrue(1, 1);
57 
58  CScript::const_iterator pc = m_Script.begin();
59  CScript::const_iterator pend = m_Script.end();
60  CScript::const_iterator pbegincodehash = m_Script.begin();
61 
62  opcodetype opcode;
63  valtype vchPushValue;
64  std::vector<bool> vfExec;
65 
66  if (m_Script.size() > MAX_SCRIPT_SIZE)
67  return set_error(serror, scriptError::SCRIPT_SIZE);
68  int nOpCount = 0;
69  bool fRequireMinimal = (flags & SCRIPT_VERIFY::MINIMALDATA) != 0;
70 
71  try
72  {
73  while (pc < pend)
74  {
75  bool fExec = !count(vfExec.begin(), vfExec.end(), false);
76 
77  //
78  // Read instruction
79  //
80  if (!m_Script.GetOp(pc, opcode, vchPushValue))
81  return set_error(serror, scriptError::BAD_OPCODE);
82  if (vchPushValue.size() > MAX_SCRIPT_ELEMENT_SIZE)
83  return set_error(serror, scriptError::PUSH_SIZE);
84 
85  // Note how OP_RESERVED does not count towards the opcode limit.
86  if (opcode > OP_16 && ++nOpCount > MAX_OPS_PER_SCRIPT)
87  return set_error(serror, scriptError::OP_COUNT);
88 
89  if (opcode == OP_CAT ||
90  opcode == OP_SUBSTR ||
91  opcode == OP_LEFT ||
92  opcode == OP_RIGHT ||
93  opcode == OP_INVERT ||
94  opcode == OP_AND ||
95  opcode == OP_OR ||
96  opcode == OP_XOR ||
97  opcode == OP_2MUL ||
98  opcode == OP_2DIV ||
99  opcode == OP_MUL ||
100  opcode == OP_DIV ||
101  opcode == OP_MOD ||
102  opcode == OP_LSHIFT ||
103  opcode == OP_RSHIFT)
104  return set_error(serror, scriptError::DISABLED_OPCODE); // Disabled opcodes.
105 
106  if (fExec && 0 <= opcode && opcode <= OP_PUSHDATA4) {
107  if (fRequireMinimal && !CheckMinimalPush(vchPushValue, opcode)) {
108  return set_error(serror, scriptError::ERR_MINIMALDATA);
109  }
110  m_vecStack.push_back(vchPushValue);
111  }
112  else if (fExec || (OP_IF <= opcode && opcode <= OP_ENDIF))
113  switch (opcode)
114  {
115  //
116  // Push value
117  //
118  case OP_1NEGATE:
119  case OP_1:
120  case OP_2:
121  case OP_3:
122  case OP_4:
123  case OP_5:
124  case OP_6:
125  case OP_7:
126  case OP_8:
127  case OP_9:
128  case OP_10:
129  case OP_11:
130  case OP_12:
131  case OP_13:
132  case OP_14:
133  case OP_15:
134  case OP_16:
135  {
136  // ( -- value)
137  CScriptNum bn((int)opcode - (int)(OP_1 - 1));
138  m_vecStack.push_back(bn.getvch());
139  // The result of these opcodes should always be the minimal way to push the data
140  // they push, so no need for a CheckMinimalPush here.
141  }
142  break;
143 
144 
145  //
146  // Control
147  //
148  case OP_NOP:
149  break;
150 
151  case OP_NOP1: case OP_NOP4: case OP_NOP5:
152  case OP_NOP6: case OP_NOP7: case OP_NOP8: case OP_NOP9: case OP_NOP10:
153  {
154  if (flags & SCRIPT_VERIFY::DISCOURAGE_UPGRADABLE_NOPS)
155  return set_error(serror, scriptError::ERR_DISCOURAGE_UPGRADABLE_NOPS);
156  }
157  break;
158 
159  case OP_IF:
160  case OP_NOTIF:
161  {
162  // <expression> if [statements] [else [statements]] endif
163  bool fValue = false;
164  if (fExec)
165  {
166  if (m_vecStack.size() < 1)
167  return set_error(serror, scriptError::UNBALANCED_CONDITIONAL);
168  valtype& vch = stacktop(-1);
169 
170  fValue = CastToBool(vch);
171  if (opcode == OP_NOTIF)
172  fValue = !fValue;
173  popStack();
174  }
175  vfExec.push_back(fValue);
176  }
177  break;
178 
179  case OP_ELSE:
180  {
181  if (vfExec.empty())
182  return set_error(serror, scriptError::UNBALANCED_CONDITIONAL);
183  vfExec.back() = !vfExec.back();
184  }
185  break;
186 
187  case OP_ENDIF:
188  {
189  if (vfExec.empty())
190  return set_error(serror, scriptError::UNBALANCED_CONDITIONAL);
191  vfExec.pop_back();
192  }
193  break;
194 
195  case OP_VERIFY:
196  {
197  // (true -- ) or
198  // (false -- false) and return
199  if (m_vecStack.size() < 1)
200  return set_error(serror, scriptError::INVALID_STACK_OPERATION);
201  bool fValue = CastToBool(stacktop(-1));
202  if (fValue)
203  popStack();
204  else
205  return set_error(serror, scriptError::VERIFY);
206  }
207  break;
208 
209  case OP_RETURN:
210  {
211  return set_error(serror, scriptError::RETURN);
212  }
213  break;
214 
215 
216  //
217  // Stack ops
218  //
219  case OP_TOALTSTACK:
220  {
221  if (m_vecStack.size() < 1)
222  return set_error(serror, scriptError::INVALID_STACK_OPERATION);
223  m_vecAltStack.push_back(stacktop(-1));
224  popStack();
225  }
226  break;
227 
228  case OP_FROMALTSTACK:
229  {
230  if (m_vecAltStack.size() < 1)
231  return set_error(serror, scriptError::INVALID_ALTSTACK_OPERATION);
232  m_vecStack.push_back(altstacktop(-1));
233  popAltStack();
234  }
235  break;
236 
237  case OP_2DROP:
238  {
239  // (x1 x2 -- )
240  if (m_vecStack.size() < 2)
241  return set_error(serror, scriptError::INVALID_STACK_OPERATION);
242  popStack();
243  popStack();
244  }
245  break;
246 
247  case OP_2DUP:
248  {
249  // (x1 x2 -- x1 x2 x1 x2)
250  if (m_vecStack.size() < 2)
251  return set_error(serror, scriptError::INVALID_STACK_OPERATION);
252  valtype vch1 = stacktop(-2);
253  valtype vch2 = stacktop(-1);
254  m_vecStack.push_back(vch1);
255  m_vecStack.push_back(vch2);
256  }
257  break;
258 
259  case OP_3DUP:
260  {
261  // (x1 x2 x3 -- x1 x2 x3 x1 x2 x3)
262  if (m_vecStack.size() < 3)
263  return set_error(serror, scriptError::INVALID_STACK_OPERATION);
264  valtype vch1 = stacktop(-3);
265  valtype vch2 = stacktop(-2);
266  valtype vch3 = stacktop(-1);
267  m_vecStack.push_back(vch1);
268  m_vecStack.push_back(vch2);
269  m_vecStack.push_back(vch3);
270  }
271  break;
272 
273  case OP_2OVER:
274  {
275  // (x1 x2 x3 x4 -- x1 x2 x3 x4 x1 x2)
276  if (m_vecStack.size() < 4)
277  return set_error(serror, scriptError::INVALID_STACK_OPERATION);
278  valtype vch1 = stacktop(-4);
279  valtype vch2 = stacktop(-3);
280  m_vecStack.push_back(vch1);
281  m_vecStack.push_back(vch2);
282  }
283  break;
284 
285  case OP_2ROT:
286  {
287  // (x1 x2 x3 x4 x5 x6 -- x3 x4 x5 x6 x1 x2)
288  if (m_vecStack.size() < 6)
289  return set_error(serror, scriptError::INVALID_STACK_OPERATION);
290  valtype vch1 = stacktop(-6);
291  valtype vch2 = stacktop(-5);
292  m_vecStack.erase(m_vecStack.end() - 6, m_vecStack.end() - 4);
293  m_vecStack.push_back(vch1);
294  m_vecStack.push_back(vch2);
295  }
296  break;
297 
298  case OP_2SWAP:
299  {
300  // (x1 x2 x3 x4 -- x3 x4 x1 x2)
301  if (m_vecStack.size() < 4)
302  return set_error(serror, scriptError::INVALID_STACK_OPERATION);
303  swap(stacktop(-4), stacktop(-2));
304  swap(stacktop(-3), stacktop(-1));
305  }
306  break;
307 
308  case OP_IFDUP:
309  {
310  // (x - 0 | x x)
311  if (m_vecStack.size() < 1)
312  return set_error(serror, scriptError::INVALID_STACK_OPERATION);
313  valtype vch = stacktop(-1);
314  if (CastToBool(vch))
315  m_vecStack.push_back(vch);
316  }
317  break;
318 
319  case OP_DEPTH:
320  {
321  // -- stacksize
322  CScriptNum bn(m_vecStack.size());
323  m_vecStack.push_back(bn.getvch());
324  }
325  break;
326 
327  case OP_DROP:
328  {
329  // (x -- )
330  if (m_vecStack.size() < 1)
331  return set_error(serror, scriptError::INVALID_STACK_OPERATION);
332  popStack();
333  }
334  break;
335 
336  case OP_DUP:
337  {
338  // (x -- x x)
339  if (m_vecStack.size() < 1)
340  return set_error(serror, scriptError::INVALID_STACK_OPERATION);
341  valtype vch = stacktop(-1);
342  m_vecStack.push_back(vch);
343  }
344  break;
345 
346  case OP_NIP:
347  {
348  // (x1 x2 -- x2)
349  if (m_vecStack.size() < 2)
350  return set_error(serror, scriptError::INVALID_STACK_OPERATION);
351  m_vecStack.erase(m_vecStack.end() - 2);
352  }
353  break;
354 
355  case OP_OVER:
356  {
357  // (x1 x2 -- x1 x2 x1)
358  if (m_vecStack.size() < 2)
359  return set_error(serror, scriptError::INVALID_STACK_OPERATION);
360  valtype vch = stacktop(-2);
361  m_vecStack.push_back(vch);
362  }
363  break;
364 
365  case OP_PICK:
366  case OP_ROLL:
367  {
368  // (xn ... x2 x1 x0 n - xn ... x2 x1 x0 xn)
369  // (xn ... x2 x1 x0 n - ... x2 x1 x0 xn)
370  if (m_vecStack.size() < 2)
371  return set_error(serror, scriptError::INVALID_STACK_OPERATION);
372  int n = CScriptNum(stacktop(-1), fRequireMinimal).getint();
373  popStack();
374  if (n < 0 || n >= (int)m_vecStack.size())
375  return set_error(serror, scriptError::INVALID_STACK_OPERATION);
376  valtype vch = stacktop(-n - 1);
377  if (opcode == OP_ROLL)
378  m_vecStack.erase(m_vecStack.end() - n - 1);
379  m_vecStack.push_back(vch);
380  }
381  break;
382 
383  case OP_ROT:
384  {
385  // (x1 x2 x3 -- x2 x3 x1)
386  // x2 x1 x3 after first swap
387  // x2 x3 x1 after second swap
388  if (m_vecStack.size() < 3)
389  return set_error(serror, scriptError::INVALID_STACK_OPERATION);
390  swap(stacktop(-3), stacktop(-2));
391  swap(stacktop(-2), stacktop(-1));
392  }
393  break;
394 
395  case OP_SWAP:
396  {
397  // (x1 x2 -- x2 x1)
398  if (m_vecStack.size() < 2)
399  return set_error(serror, scriptError::INVALID_STACK_OPERATION);
400  swap(stacktop(-2), stacktop(-1));
401  }
402  break;
403 
404  case OP_TUCK:
405  {
406  // (x1 x2 -- x2 x1 x2)
407  if (m_vecStack.size() < 2)
408  return set_error(serror, scriptError::INVALID_STACK_OPERATION);
409  valtype vch = stacktop(-1);
410  m_vecStack.insert(m_vecStack.end() - 2, vch);
411  }
412  break;
413 
414 
415  case OP_SIZE:
416  {
417  // (in -- in size)
418  if (m_vecStack.size() < 1)
419  return set_error(serror, scriptError::INVALID_STACK_OPERATION);
420  CScriptNum bn(stacktop(-1).size());
421  m_vecStack.push_back(bn.getvch());
422  }
423  break;
424 
425 
426  //
427  // Bitwise logic
428  //
429  case OP_EQUAL:
430  case OP_EQUALVERIFY:
431  //case OP_NOTEQUAL: // use OP_NUMNOTEQUAL
432  {
433  // (x1 x2 - bool)
434  if (m_vecStack.size() < 2)
435  return set_error(serror, scriptError::INVALID_STACK_OPERATION);
436  valtype& vch1 = stacktop(-2);
437  valtype& vch2 = stacktop(-1);
438  bool fEqual = (vch1 == vch2);
439  // OP_NOTEQUAL is disabled because it would be too easy to say
440  // something like n != 1 and have some wiseguy pass in 1 with extra
441  // zero bytes after it (numerically, 0x01 == 0x0001 == 0x000001)
442  //if (opcode == OP_NOTEQUAL)
443  // fEqual = !fEqual;
444  popStack();
445  popStack();
446  m_vecStack.push_back(fEqual ? vchTrue : vchFalse);
447  if (opcode == OP_EQUALVERIFY)
448  {
449  if (fEqual)
450  popStack();
451  else
452  return set_error(serror, scriptError::EQUALVERIFY);
453  }
454  }
455  break;
456 
457 
458  //
459  // Numeric
460  //
461  case OP_1ADD:
462  case OP_1SUB:
463  case OP_NEGATE:
464  case OP_ABS:
465  case OP_NOT:
466  case OP_0NOTEQUAL:
467  {
468  // (in -- out)
469  if (m_vecStack.size() < 1)
470  return set_error(serror, scriptError::INVALID_STACK_OPERATION);
471  CScriptNum bn(stacktop(-1), fRequireMinimal);
472  switch (opcode)
473  {
474  case OP_1ADD: bn += bnOne; break;
475  case OP_1SUB: bn -= bnOne; break;
476  case OP_NEGATE: bn = -bn; break;
477  case OP_ABS: if (bn < bnZero) bn = -bn; break;
478  case OP_NOT: bn = (bn == bnZero); break;
479  case OP_0NOTEQUAL: bn = (bn != bnZero); break;
480  default: assert(!"invalid opcode"); break;
481  }
482  popStack();
483  m_vecStack.push_back(bn.getvch());
484  }
485  break;
486 
487  case OP_ADD:
488  case OP_SUB:
489  case OP_BOOLAND:
490  case OP_BOOLOR:
491  case OP_NUMEQUAL:
492  case OP_NUMEQUALVERIFY:
493  case OP_NUMNOTEQUAL:
494  case OP_LESSTHAN:
495  case OP_GREATERTHAN:
496  case OP_LESSTHANOREQUAL:
497  case OP_GREATERTHANOREQUAL:
498  case OP_MIN:
499  case OP_MAX:
500  {
501  // (x1 x2 -- out)
502  if (m_vecStack.size() < 2)
503  return set_error(serror, scriptError::INVALID_STACK_OPERATION);
504  CScriptNum bn1(stacktop(-2), fRequireMinimal);
505  CScriptNum bn2(stacktop(-1), fRequireMinimal);
506  CScriptNum bn(0);
507  switch (opcode)
508  {
509  case OP_ADD:
510  bn = bn1 + bn2;
511  break;
512 
513  case OP_SUB:
514  bn = bn1 - bn2;
515  break;
516 
517  case OP_BOOLAND: bn = (bn1 != bnZero && bn2 != bnZero); break;
518  case OP_BOOLOR: bn = (bn1 != bnZero || bn2 != bnZero); break;
519  case OP_NUMEQUAL: bn = (bn1 == bn2); break;
520  case OP_NUMEQUALVERIFY: bn = (bn1 == bn2); break;
521  case OP_NUMNOTEQUAL: bn = (bn1 != bn2); break;
522  case OP_LESSTHAN: bn = (bn1 < bn2); break;
523  case OP_GREATERTHAN: bn = (bn1 > bn2); break;
524  case OP_LESSTHANOREQUAL: bn = (bn1 <= bn2); break;
525  case OP_GREATERTHANOREQUAL: bn = (bn1 >= bn2); break;
526  case OP_MIN: bn = (bn1 < bn2 ? bn1 : bn2); break;
527  case OP_MAX: bn = (bn1 > bn2 ? bn1 : bn2); break;
528  default: assert(!"invalid opcode"); break;
529  }
530  popStack();
531  popStack();
532  m_vecStack.push_back(bn.getvch());
533 
534  if (opcode == OP_NUMEQUALVERIFY)
535  {
536  if (CastToBool(stacktop(-1)))
537  popStack();
538  else
539  return set_error(serror, scriptError::NUMEQUALVERIFY);
540  }
541  }
542  break;
543 
544  case OP_WITHIN:
545  {
546  // (x min max -- out)
547  if (m_vecStack.size() < 3)
548  return set_error(serror, scriptError::INVALID_STACK_OPERATION);
549  CScriptNum bn1(stacktop(-3), fRequireMinimal);
550  CScriptNum bn2(stacktop(-2), fRequireMinimal);
551  CScriptNum bn3(stacktop(-1), fRequireMinimal);
552  bool fValue = (bn2 <= bn1 && bn1 < bn3);
553  popStack();
554  popStack();
555  popStack();
556  m_vecStack.push_back(fValue ? vchTrue : vchFalse);
557  }
558  break;
559 
560 
561  //
562  // Crypto
563  //
564  case OP_RIPEMD160:
565  case OP_SHA1:
566  case OP_SHA256:
567  case OP_HASH256:
568  case OP_SHA3_256:
569  case OP_SHA3_512:
570  case OP_SHA3_CSHAKE:
571  {
572  // (in -- hash)
573  if (m_vecStack.size() < 1)
574  return set_error(serror, scriptError::INVALID_STACK_OPERATION);
575  valtype& vch = stacktop(-1);
576  valtype vchHash((opcode == OP_RIPEMD160 || opcode == OP_SHA1 ) ? 20 : 32);
577  if (opcode == OP_RIPEMD160)
578  CRIPEMD160().Write(vch.data(), vch.size()).Finalize(vchHash.data());
579  else if (opcode == OP_SHA1)
580  CSHA1().Write(vch.data(), vch.size()).Finalize(vchHash.data());
581  else if (opcode == OP_SHA256)
582  CSHA256().Write(vch.data(), vch.size()).Finalize(vchHash.data());
583  else if (opcode == OP_HASH256)
584  cHash256().Write(vch.data(), vch.size()).Finalize(vchHash.data());
585 
586  // todo: insert sha3 methos
587 
588  popStack();
589  m_vecStack.push_back(vchHash);
590  }
591  break;
592 
593  case OP_CODESEPARATOR:
594  {
595  // Hash starts after the code separator
596  pbegincodehash = pc;
597  }
598  break;
599 
600  case OP_CHECKSIG:
601  case OP_CHECKSIGVERIFY:
602  {
603  // (sig pubkey -- bool)
604  if (m_vecStack.size() < 2)
605  return set_error(serror, scriptError::INVALID_STACK_OPERATION);
606 
607  valtype& vchSig = stacktop(-2);
608  valtype& vchPubKey = stacktop(-1);
609 
610  // Subset of script starting at the most recent codeseparator
611  CScript scriptCode(pbegincodehash, pend);
612 
613  // Drop the signature
614  scriptCode.FindAndDelete(CScript(vchSig));
615 
616  if (!CheckSignatureEncoding(vchSig, flags, serror) || !CheckPubKeyEncoding(vchPubKey, flags, serror))
617  {
618  //serror is set
619  return false;
620  }
621  bool fSuccess = m_signatureChecker.checkSignature(vchSig, vchPubKey, scriptCode);
622 
623  if (!fSuccess && (flags & SCRIPT_VERIFY::NULLFAIL) && vchSig.size())
624  return set_error(serror, scriptError::SIG_NULLFAIL);
625 
626  popStack();
627  popStack();
628  m_vecStack.push_back(fSuccess ? vchTrue : vchFalse);
629  if (opcode == OP_CHECKSIGVERIFY)
630  {
631  if (fSuccess)
632  popStack();
633  else
634  return set_error(serror, scriptError::CHECKSIGVERIFY);
635  }
636  }
637  break;
638 
639  case OP_CHECKMULTISIG:
640  case OP_CHECKMULTISIGVERIFY:
641  {
642  // ([sig ...] num_of_signatures [pubkey ...] num_of_pubkeys -- bool)
643 
644  int i = 1;
645  if ((int)m_vecStack.size() < i)
646  return set_error(serror, scriptError::INVALID_STACK_OPERATION);
647 
648  int nKeysCount = CScriptNum(stacktop(-i), fRequireMinimal).getint();
649  if (nKeysCount < 0 || nKeysCount > MAX_PUBKEYS_PER_MULTISIG)
650  return set_error(serror, scriptError::PUBKEY_COUNT);
651  nOpCount += nKeysCount;
652  if (nOpCount > MAX_OPS_PER_SCRIPT)
653  return set_error(serror, scriptError::OP_COUNT);
654  int ikey = ++i;
655  // ikey2 is the position of last non-signature item in the m_vecStack. Top stack item = 1.
656  // With SCRIPT_VERIFY::NULLFAIL, this is used for cleanup if operation fails.
657  int ikey2 = nKeysCount + 2;
658  i += nKeysCount;
659  if ((int)m_vecStack.size() < i)
660  return set_error(serror, scriptError::INVALID_STACK_OPERATION);
661 
662  int nSigsCount = CScriptNum(stacktop(-i), fRequireMinimal).getint();
663  if (nSigsCount < 0 || nSigsCount > nKeysCount)
664  return set_error(serror, scriptError::SIG_COUNT);
665  int isig = ++i;
666  i += nSigsCount;
667  if ((int)m_vecStack.size() < i)
668  return set_error(serror, scriptError::INVALID_STACK_OPERATION);
669 
670  // Subset of script starting at the most recent codeseparator
671  CScript scriptCode(pbegincodehash, pend);
672 
673  // Drop the signature in pre-segwit scripts but not segwit scripts
674  for (int k = 0; k < nSigsCount; k++)
675  {
676  valtype& vchSig = stacktop(-isig - k);
677  scriptCode.FindAndDelete(CScript(vchSig));
678  }
679 
680  bool fSuccess = true;
681  while (fSuccess && nSigsCount > 0)
682  {
683  valtype& vchSig = stacktop(-isig);
684  valtype& vchPubKey = stacktop(-ikey);
685 
686  // Note how this makes the exact order of pubkey/signature evaluation
687  // distinguishable by CHECKMULTISIG NOT if the STRICTENC flag is set.
688  // See the script_(in)valid tests for details.
689  if (!CheckSignatureEncoding(vchSig, flags, serror) || !CheckPubKeyEncoding(vchPubKey, flags, serror)) {
690  // serror is set
691  return false;
692  }
693 
694  // Check signature
695  bool fOk = m_signatureChecker.checkSignature(vchSig, vchPubKey, scriptCode);
696 
697  if (fOk) {
698  isig++;
699  nSigsCount--;
700  }
701  ikey++;
702  nKeysCount--;
703 
704  // If there are more signatures left than keys left,
705  // then too many signatures have failed. Exit early,
706  // without checking any further signatures.
707  if (nSigsCount > nKeysCount)
708  fSuccess = false;
709  }
710 
711  // Clean up stack of actual arguments
712  while (i-- > 1) {
713  // If the operation failed, we require that all signatures must be empty vector
714  if (!fSuccess && (flags & SCRIPT_VERIFY::NULLFAIL) && !ikey2 && stacktop(-1).size())
715  return set_error(serror, scriptError::SIG_NULLFAIL);
716  if (ikey2 > 0)
717  ikey2--;
718  popStack();
719  }
720 
721  // A bug causes CHECKMULTISIG to consume one extra argument
722  // whose contents were not checked in any way.
723  //
724  // Unfortunately this is a potential source of mutability,
725  // so optionally verify it is exactly equal to zero prior
726  // to removing it from the m_vecStack.
727  if (m_vecStack.size() < 1)
728  return set_error(serror, scriptError::INVALID_STACK_OPERATION);
729  if ((flags & SCRIPT_VERIFY::NULLDUMMY) && stacktop(-1).size())
730  return set_error(serror, scriptError::SIG_NULLDUMMY);
731  popStack();
732 
733  m_vecStack.push_back(fSuccess ? vchTrue : vchFalse);
734 
735  if (opcode == OP_CHECKMULTISIGVERIFY)
736  {
737  if (fSuccess)
738  popStack();
739  else
740  return set_error(serror, scriptError::CHECKMULTISIGVERIFY);
741  }
742  }
743  break;
744 
745  default:
746  return set_error(serror, scriptError::BAD_OPCODE);
747  }
748 
749  // Size limits
750  if (m_vecStack.size() + m_vecAltStack.size() > MAX_STACK_SIZE)
751  return set_error(serror, scriptError::STACK_SIZE);
752  }
753  }
754  catch (...)
755  {
756  return set_error(serror, scriptError::UNKNOWN_ERROR);
757  }
758 
759  if (!vfExec.empty())
760  return set_error(serror, scriptError::UNBALANCED_CONDITIONAL);
761 
762  return set_success(serror);
763  }

766  bool scriptInterpreter::verifyScript( const CScript& scriptPubKey, SCRIPT_VERIFY flags, scriptError* serror)
767  {
768  if ((flags & SCRIPT_VERIFY::SIGPUSHONLY) != 0 && !m_Script.IsPushOnly()) {
769  return set_error(serror, scriptError::SIG_PUSHONLY);
770  }
771 
772  std::vector<std::vector<unsigned char> > vecStackCopy;
773  if (!evalScript( flags, serror))
774  // serror is set
775  return false;
776  if (flags & SCRIPT_VERIFY::P2SH)
777  vecStackCopy = m_vecStack;
778 
779  // eval the pubkey
780  CScript scriptBackup = m_Script;
781  m_Script = scriptPubKey;
782  if (!evalScript( flags, serror))
783  // serror is set
784  return false;
785  m_Script = scriptBackup;
786 
787  if (m_vecStack.empty())
788  return set_error(serror, scriptError::EVAL_FALSE);
789  if (CastToBool(m_vecStack.back()) == false)
790  return set_error(serror, scriptError::EVAL_FALSE);
791 
792  // Additional validation for spend-to-script-hash transactions:
793  if ((flags & SCRIPT_VERIFY::P2SH) && scriptPubKey.IsPayToScriptHash())
794  {
795  // scriptSig must be literals-only or validation fails
796  if (!m_Script.IsPushOnly())
797  return set_error(serror, scriptError::SIG_PUSHONLY);
798 
799  // Restore stack.
800  swap(m_vecStack, vecStackCopy);
801 
802  // stack cannot be empty here, because if it was the
803  // P2SH HASH <> EQUAL scriptPubKey would be evaluated with
804  // an empty stack and the EvalScript above would return false.
805  if (m_vecStack.empty())
806  {
807  LOG_ERROR("Stack empty", "scriptInterpreter");
808  return set_error(serror, scriptError::INVALID_STACK_OPERATION);
809  }
810 
811  const valtype& pubKeySerialized = m_vecStack.back();
812  CScript pubKey2(pubKeySerialized.begin(), pubKeySerialized.end());
813  popStack();
814 
815  scriptBackup = m_Script;
816  m_Script = pubKey2;
817  if (!evalScript( flags, serror))
818  // serror is set
819  return false;
820  m_Script = scriptBackup;
821 
822  if (m_vecStack.empty())
823  return set_error(serror, scriptError::EVAL_FALSE);
824  if (!CastToBool(m_vecStack.back()))
825  return set_error(serror, scriptError::EVAL_FALSE);
826  }
827 
828  // The CLEANSTACK check is only performed after potential P2SH evaluation,
829  // as the non-P2SH evaluation of a P2SH script will obviously not result in
830  // a clean stack (the P2SH inputs remain). The same holds for witness evaluation.
831  if ((flags & SCRIPT_VERIFY::CLEANSTACK) != 0)
832  {
833  // Disallow CLEANSTACK without P2SH, as otherwise a switch CLEANSTACK->P2SH+CLEANSTACK
834  // would be possible, which is not a softfork (and P2SH should be one).
835  if ((flags & SCRIPT_VERIFY::P2SH) != 0)
836  {
837  LOG_ERROR("SCRIPT_VERIFY::CLEANSTACK can't run without SCRIPT_VERIFY::P2SH", "scriptInterpreter");
838  return set_error(serror, scriptError::ERR_CLEANSTACK);
839  }
840  if (m_vecStack.size() != 1)
841  return set_error(serror, scriptError::ERR_CLEANSTACK);
842  }
843 
844  return set_success(serror);
845  }
