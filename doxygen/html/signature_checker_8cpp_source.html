49  bool signatureChecker::checkLockTime(const CScriptNum& nLockTime)
50  {
51  // There are two kinds of nLockTime: lock-by-blockheight
52  // and lock-by-blocktime, distinguished by whether
53  // nLockTime < LOCKTIME_THRESHOLD.
54  //
55  // We want to compare apples to apples, so fail the script
56  // unless the type of nLockTime being tested is the same as
57  // the nLockTime in the transaction.
58  if (!(
59  (txTo->uint32tLockTime < LOCKTIME_THRESHOLD && nLockTime < LOCKTIME_THRESHOLD) ||
60  (txTo->uint32tLockTime >= LOCKTIME_THRESHOLD && nLockTime >= LOCKTIME_THRESHOLD)
61  ))
62  return false;
63 
64  // Now that we know we're comparing apples-to-apples, the
65  // comparison is a simple numeric one.
