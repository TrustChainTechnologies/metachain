111 int netMessage::readData(const char *pch, unsigned int nBytes)
112 {
113  if (!m_bHeaderComplete)
114  {
115  // read the header first
116  unsigned int uiRemaining = sizeof(sHeader) - m_uiPosition;
117  unsigned int uiBufSize = (std::min)(uiRemaining, nBytes);
118 
119  // buffer allocation, reading and freeing
120  char *pHeaderPos = (char *)(&m_sHeader) + m_uiPosition;
121  memcpy(pHeaderPos, pch, uiBufSize);
122  m_uiPosition += uiBufSize;
123 
124 #ifdef _DEBUG
125  LOG_DEBUG("Reading Header - Header Position: " + std::to_string(m_uiPosition) + ", Bytes read: " + std::to_string(uiBufSize), "NET-MSG");
126 #endif
127 
128  // if the header is not fully read, we return the number of bytes read
129  if (m_uiPosition < sizeof(sHeader))
130  return uiBufSize;
131 
132  // when we reach this point the header is fully read, we sanitize everything for data reading
133  m_bHeaderComplete = true;
134  m_uiPosition = 0;
135 
136  // security check for payload DoS
137  if (m_sHeader.ui32tPayloadSize > MAX_PAYLOAD_SIZE)
138  {
139  LOG_ERROR("DoS data package filtered, disconnecting node", "NET-MSG");
140  return -1;
141  }
142 
143  // allocating the buffer for the data
144  m_pBuffer = new char[sizeof(sHeader) + m_sHeader.ui32tPayloadSize];
145 
146  // copying the header into the buffer
147  memcpy(m_pBuffer, &m_sHeader, sizeof(sHeader));
148 
149 #ifdef _DEBUG
150  uint256 tmpCmp;
151  memcpy(tmpCmp.begin(), m_sHeader.ui8tChecksum, CHECKSUM_SIZE );
152  LOG_DEBUG("Header Completed - Subject: " + std::to_string(m_sHeader.ui16tSubject) + ", Payload Size: " + std::to_string(m_sHeader.ui32tPayloadSize) + ", Header Hash: " + tmpCmp.ToString(), "NET-MSG");
153 #endif
154 
155  // when we don't have any payload after the header, the message is complete
156  if (m_sHeader.ui32tPayloadSize == 0)
157  m_bComplete = true;
158 
159  // everything smooth, return the bytes read
160  return uiBufSize;
161  }
162  else
163  {
164  // calculate the remaining bytes of the message as well as the number of bytes to read
165  unsigned int uiRemaining = m_sHeader.ui32tPayloadSize - m_uiPosition;
166  unsigned int uiBufSize = (std::min)(uiRemaining, nBytes);
167 
168  memcpy((m_pBuffer + sizeof(sHeader) + m_uiPosition), pch, uiBufSize);
169  m_uiPosition += uiBufSize;
170  m_Hasher256.Write((const unsigned char*)pch, uiBufSize);
171 
172 #ifdef _DEBUG
173  LOG_DEBUG("Reading Data - Data Position: " + std::to_string(m_uiPosition) + ", Bytes read: " + std::to_string(uiBufSize), "NET-MSG");
174 #endif
175 
176  // check wether we completed the data part
177  if (m_sHeader.ui32tPayloadSize == m_uiPosition)
178  {
179  m_bComplete = true;
180  GetMessageHash();
181 #ifdef _DEBUG
182  uint256 tmpHead, tmpCalc;
183  memcpy(tmpHead.begin(), m_sHeader.ui8tChecksum, CHECKSUM_SIZE );
184  memcpy(tmpCalc.begin(), m_ui256DataHash.begin(), CHECKSUM_SIZE);
185  LOG_DEBUG("Checking Hashsums - Header hash: " + tmpHead.ToString() + ", calculated data hash: " + tmpCalc.ToString(), "NET-MSG");
186 #endif
187 
188  if (memcmp(m_ui256DataHash.begin(), m_sHeader.ui8tChecksum, CHECKSUM_SIZE) != 0)
189  {
190  LOG_ERROR("Received package with invalid hash, disconnecting node", "NET-MSG");
191  return -1;
192  }
193 #ifdef _DEBUG
194  else
195  LOG_DEBUG("Header checksum matches", "NET-MSG");
196 #endif
197  }
198 
199  return uiBufSize;
200  }
