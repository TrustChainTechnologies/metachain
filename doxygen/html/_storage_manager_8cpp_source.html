190 bool StorageManager::openRawFile(unsigned short usChainIdentifier)
191 {
192  // get an iterator for faster access
193  auto it = m_umapSC.find(usChainIdentifier);
194  if (it == m_umapSC.end())
195  {
196  LOG_ERROR("Can't find entry to open raw file for SC: " + std::to_string(usChainIdentifier), "SM");
197  return false;
198  }
199 
200  // close the stream if opened
201  LOCK(it->second.csAccess);
202  if (it->second.streamRaw.is_open())
203  it->second.streamRaw.close();
204 
205  // open our fstream for raw output
206  std::ostringstream ssFileName;
207  ssFileName << std::setw(8) << std::setfill('0') << it->second.uiRawFileCounter;
208  it->second.filePath = m_pathRawDirectory / std::to_string(usChainIdentifier) / (ssFileName.str() + RAW_FILEENDING);
209 
210  // if the file exists we get the size to calculate our splitting. If it doesn't we start with 0
211  if (boost::filesystem::exists(it->second.filePath))
212  it->second.uimRawFileSize = boost::filesystem::file_size(it->second.filePath);
213  else
214  it->second.uimRawFileSize = 0;
215 
216  it->second.streamRaw.open(it->second.filePath.string(), std::ios_base::app);
217  if (!it->second.streamRaw.is_open())
218  {
219  LOG_ERROR("Unable to open raw file: " + it->second.filePath.string(), "SM");
220  return false;
221  }
222 
223  return true;
224 }

226 bool StorageManager::writeRaw(unsigned short usChainIdentifier, unsigned int uiLength, void *raw)
227 {
228  // get the SC dbEngine
229  dbEngine *db = m_pSubChainManager->getDBEngine(usChainIdentifier);
230  if (db == nullptr)
231  {
232  LOG_ERROR("Can't open db for SC to store information: " + std::to_string(usChainIdentifier), "SM");
233  return false;
234  }
235 
236  // get the last block number and increment it to our current block
237  unsigned int uiBlockNumber = db->get("last_block", (unsigned int)0) + 1;
238 
239  if (m_pMC->isFN())
240  {
241  // get an iterator for faster access
242  auto it = m_umapSC.find(usChainIdentifier);
243  if (it == m_umapSC.end())
244  {
245  LOG_ERROR("Can't find entry to store raw block for SC: " + std::to_string(usChainIdentifier), "SM");
246  return false;
247  }
248 
249  // check whether we have an open stream or not
250  if (!it->second.streamRaw.is_open())
251  {
252  if (!openRawFile(usChainIdentifier))
253  {
254  LOG_ERROR("Can't open raw stream for output: " + std::to_string(usChainIdentifier), "SM");
255  return false;
256  }
257  }
258 
259  // write the output in the file
260  {
261  LOCK(it->second.csAccess);
262  it->second.streamRaw.write((char *)raw, uiLength);
263  it->second.streamRaw.flush();
264  it->second.uimRawFileSize += uiLength;
265  }
266 
267  // write some meta data
268  db->batchStart();
269  db->batchAddStatement( "mbl." + std::to_string(uiBlockNumber) + ".file", it->second.filePath.string());
270  db->batchAddStatement( "mbl." + std::to_string(uiBlockNumber) + ".offset", std::to_string(it->second.uimRawFileSize));
271  db->batchAddStatement( "mbl." + std::to_string(uiBlockNumber) + ".length", std::to_string(uiLength));
272  db->batchFinalize();
273 
274  // if the new file size is bigger than our maximum value, we close this file and open a new one for output
275  if (it->second.uimRawFileSize >= m_uimRawFileSizeMax)
276  {
277  LOCK(it->second.csAccess);
278  it->second.uiRawFileCounter++;
279 
280  if (!openRawFile(usChainIdentifier))
281  return false;
282  }
283  }
284 
285  // write the current block height into the db
286  db->write("last_block", std::to_string(uiBlockNumber));
287 
288  return true;
289 }

302 bool StorageManager::registerSC(unsigned short usChainIdentifier)
303 {
304  LOCK(m_csUmapSC);
305 
306  // check if the SC already is registered
307  if (m_umapSC.count(usChainIdentifier) != 0)
308  {
309  LOG_ERROR("Can't register SC since it already exists: " + usChainIdentifier, "SM");
310  return false;
311  }
312 
313  // push back the new SC into our map (using [] on not created element will create a new element)
314  m_umapSC[usChainIdentifier].uimRawFileSize = 0;
315  m_umapSC[usChainIdentifier].uiRawFileCounter = 0;
316 
317  // check for the raw directory (this is only needed when we're in FN mode)
318  if (m_pMC->isFN())
319  {
320  boost::filesystem::path pathRaw = m_pathRawDirectory / std::to_string(usChainIdentifier);
321  if (!boost::filesystem::exists(pathRaw) || !boost::filesystem::is_directory(pathRaw))
322  {
323  LOG("Creating Raw Directory: " + pathRaw.string(), "SM");
324  if (!boost::filesystem::create_directory(pathRaw))
325  {
326  LOG_ERROR("Couldn't create raw directory: " + pathRaw.string(), "SM");
327  return false;
328  }
329  }
330 
331  // open the raw file and initialize the streams for writing
332  openRawFile(usChainIdentifier);
333  }
334 
335  return true;
