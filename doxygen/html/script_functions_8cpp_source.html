77  bool IsValidSignatureEncoding(const std::vector<unsigned char> &sig)
78  {
79  // Format: 0x30 [total-length] 0x02 [R-length] [R] 0x02 [S-length] [S] [sighash]
80  // * total-length: 1-byte length descriptor of everything that follows,
81  // excluding the sighash byte.
82  // * R-length: 1-byte length descriptor of the R value that follows.
83  // * R: arbitrary-length big-endian encoded R value. It must use the shortest
84  // possible encoding for a positive integers (which means no null bytes at
85  // the start, except a single one when the next byte has its highest bit set).
86  // * S-length: 1-byte length descriptor of the S value that follows.
87  // * S: arbitrary-length big-endian encoded S value. The same rules apply.
88  // * sighash: 1-byte value indicating what data is hashed (not part of the DER
89  // signature)
90 
91  // Minimum and maximum size constraints.
92  if (sig.size() < 9) return false;
93  if (sig.size() > 73) return false;
94 
95  // A signature is of type 0x30 (compound).
96  if (sig[0] != 0x30) return false;
97 
98  // Make sure the length covers the entire signature.
99  if (sig[1] != sig.size() - 3) return false;
100 
101  // Extract the length of the R element.
102  unsigned int lenR = sig[3];
103 
104  // Make sure the length of the S element is still inside the signature.
105  if (5 + lenR >= sig.size()) return false;
106 
107  // Extract the length of the S element.
108  unsigned int lenS = sig[5 + lenR];
109 
110  // Verify that the length of the signature matches the sum of the length
111  // of the elements.
112  if ((size_t)(lenR + lenS + 7) != sig.size()) return false;
113 
114  // Check whether the R element is an integer.
115  if (sig[2] != 0x02) return false;
116 
117  // Zero-length integers are not allowed for R.
118  if (lenR == 0) return false;
119 
120  // Negative numbers are not allowed for R.
121  if (sig[4] & 0x80) return false;
122 
123  // Null bytes at the start of R are not allowed, unless R would
124  // otherwise be interpreted as a negative number.
125  if (lenR > 1 && (sig[4] == 0x00) && !(sig[5] & 0x80)) return false;
126 
127  // Check whether the S element is an integer.
128  if (sig[lenR + 4] != 0x02) return false;
129 
130  // Zero-length integers are not allowed for S.
131  if (lenS == 0) return false;
132 
133  // Negative numbers are not allowed for S.
134  if (sig[lenR + 6] & 0x80) return false;
135 
136  // Null bytes at the start of S are not allowed, unless S would otherwise be
137  // interpreted as a negative number.
138  if (lenS > 1 && (sig[lenR + 6] == 0x00) && !(sig[lenR + 7] & 0x80)) return false;
139 
140  return true;
141  }

143  bool IsCompressedOrUncompressedPubKey(const valtype &vchPubKey)
144  {
145  if (vchPubKey.size() < 33) {
146  // Non-canonical public key: too short
147  return false;
148  }
149  if (vchPubKey[0] == 0x04) {
150  if (vchPubKey.size() != 65) {
151  // Non-canonical public key: invalid length for uncompressed key
152  return false;
153  }
154  }
155  else if (vchPubKey[0] == 0x02 || vchPubKey[0] == 0x03) {
156  if (vchPubKey.size() != 33) {
157  // Non-canonical public key: invalid length for compressed key
158  return false;
159  }
160  }
161  else {
162  // Non-canonical public key: neither compressed nor uncompressed
163  return false;
164  }
165  return true;
166  }

201  bool CheckMinimalPush(const valtype& data, opcodetype opcode)
202  {
203  if (data.size() == 0) {
204  // Could have used OP_0.
205  return opcode == OP_0;
206  }
207  else if (data.size() == 1 && data[0] >= 1 && data[0] <= 16) {
208  // Could have used OP_1 .. OP_16.
209  return opcode == OP_1 + (data[0] - 1);
210  }
211  else if (data.size() == 1 && data[0] == 0x81) {
212  // Could have used OP_1NEGATE.
213  return opcode == OP_1NEGATE;
214  }
215  else if (data.size() <= 75) {
216  // Could have used a direct push (opcode indicating number of bytes pushed + those bytes).
217  return opcode == data.size();
218  }
219  else if (data.size() <= 255) {
220  // Could have used OP_PUSHDATA.
221  return opcode == OP_PUSHDATA1;
222  }
223  else if (data.size() <= 65535) {
224  // Could have used OP_PUSHDATA2.
225  return opcode == OP_PUSHDATA2;
226  }
227  return true;
228  }
