156  unsigned int uiBlockSize = 200 - 2 * (size / 8);
157  unsigned int uiBufferSize = (strFunctionName.length() + strCustomization.length() == 0 ? 1 : ceil((float)(strFunctionName.length() + strCustomization.length()) / uiBlockSize))*uiBlockSize + uiLength + 1;
158  m_pEncBuf = new unsigned char[uiBufferSize];
159  memset(m_pEncBuf, 0x00, uiBufferSize);

167  memcpy((m_pEncBuf + uiOffset), pPtr, uiBytesWritten);
168  uiOffset += uiBytesWritten;
169  delete pPtr;
170 
171  // the customization string
172  pPtr = encode_string((unsigned char*)strCustomization.c_str(), strCustomization.length(), &uiBytesWritten);
173  memcpy((m_pEncBuf + uiOffset), pPtr, uiBytesWritten);
174  uiOffset += uiBytesWritten;
175  delete pPtr;
176 
177  // the content itself
178  uiOffset = (strFunctionName.length() + strCustomization.length() == 0 ? 1 : ceil((float)(strFunctionName.length() + strCustomization.length()) / uiBlockSize))*uiBlockSize; //bytepad
179  memcpy((m_pEncBuf + uiOffset), pBuffer, uiLength);
180 
181  uiOffset += uiLength;

183  // call the has function with the prepared cShake buffer

258 // Process a larger buffer with varying amounts of data in it
259 void SHA3::keccakUpdate(const uint8_t *input, int off, unsigned int len)
260 {
261  uint8_t *buffer = m_keccakState.buffer;
262  while (len > 0)
263  {
264  int cpLen = 0;
265  if ((m_keccakState.blockLen - m_keccakState.bufferLen) > len)
266  {
267  cpLen = len;
268  }
269  else
270  {
271  cpLen = m_keccakState.blockLen - m_keccakState.bufferLen;
272  }
273 
274  for (unsigned int i = 0; i != cpLen; i++)
275  {
276  buffer[m_keccakState.bufferLen + i] = input[off + i];
277  }
278  m_keccakState.bufferLen += cpLen;
279  off += cpLen;
280  len -= cpLen;
281  if (m_keccakState.bufferLen == m_keccakState.blockLen)
282  {
283  keccakProcessBuffer();
284  }
285  }
