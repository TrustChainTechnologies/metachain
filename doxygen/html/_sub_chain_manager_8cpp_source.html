238  bool SubChainManager::isSubChainAllowed(std::string strChainName)
239  {
240  // first check blacklist
241  for (auto &it : MetaChain::getInstance().vecSC_Blacklist)
242  {
243 #ifdef _DEBUG
244  LOG_DEBUG("Checking Subchain Blacklist Entry: " + boost::trim_copy(it), "SCM");
245 #endif
246  // wildcard blacklist means we don't allow anything
247  if (boost::trim_copy(it) == "*")
248  return false;
249 
250  // the chain name was specified in the blacklist
251  if (boost::trim_copy(it) == strChainName)
252  return false;
253  }
254 
255  // then check whitelist
256  for (auto &it : MetaChain::getInstance().vecSC_Whitelist)
257  {
258 #ifdef _DEBUG
259  LOG_DEBUG("Checking Subchain Whitelist Entry: " + it, "SCM");
260 #endif
261  // wildcard whitelist means we allow everything
262  if (boost::trim_copy(it) == "*")
263  return true;
264 
265  // the chain name was specified in the whitelist
266  if (boost::trim_copy(it) == strChainName)
267  return true;
268  }
269 
270  // if we reached this point it means we didnt match any white or blacklist entries, therefore we're rejecting this subchain per default
271  return false;
272  }

275  uint16_t SubChainManager::addSubChain(MCP03::MetaChain::mcBlock *block)
276  {
277  // checking if this is really a genesis request block
278  if( block->pTransaction->txIn.eAction != MCP03::MetaChain::mcTxIn::ACTION::CREATE_SUBCHAIN )
279  {
280  LOG_ERROR("A non-genesis creation block was passed into the genesis creation function. Not executing!", "SCM");
281  return (std::numeric_limits<uint16_t>::max)();
282  }
283 
284  // check if this subchain is allowed in the config
285  if (!isSubChainAllowed(((MCP03::MetaChain::createSubchain*)block->pTransaction->txIn.pPayload)->caChainName))
286  {
287  LOG_ERROR("SC was not added since the configuration (subchain_whitelist, subchain_blacklist) prohibits this SC.", "SCM");
288  return (std::numeric_limits<uint16_t>::max)();
289  }
290 
291  // check whether this chain name already exists. If so, we don't add it
292  if (getChainIdentifier(((MCP03::MetaChain::createSubchain*)block->pTransaction->txIn.pPayload)->caChainName) != (std::numeric_limits<uint16_t>::max)())
293  {
294  LOG_ERROR("SC with the same name '" + std::string(((MCP03::MetaChain::createSubchain*)block->pTransaction->txIn.pPayload)->caChainName) + "' already exists. Not adding!", "SCM");
295  return (std::numeric_limits<uint16_t>::max)();
296  }
297 
298  // we also need to check whether the class for this SC exists.
299  if( !scExists(((MCP03::MetaChain::createSubchain*)block->pTransaction->txIn.pPayload)->caSubChainClassName) )
300  {
301  LOG_ERROR("The requested SC class '" + std::string(((MCP03::MetaChain::createSubchain*)block->pTransaction->txIn.pPayload)->caSubChainClassName) + "' isn't loaded. This means that this node can't participate in the new SC.", "SCM");
302  LOG_ERROR("If you wish to proceed participating in this subchain, please load the requested module and resync the MC", "SCM");
303  return (std::numeric_limits<uint16_t>::max)();
304  }
305 
306  // now we make sure that we have the requested PoP present.
307  if( !popExists(((MCP03::MetaChain::createSubchain*)block->pTransaction->txIn.pPayload)->caPoP) )
308  {
309  LOG_ERROR("The requested PoP '" + std::string(((MCP03::MetaChain::createSubchain*)block->pTransaction->txIn.pPayload)->caPoP) + "' for this SC isn't loaded. This means that this node can't participate in the new SC.", "SCM");
310  LOG_ERROR("If you wish to proceed participating in this subchain, please load the requested module and resync the MC", "SCM");
311  return (std::numeric_limits<uint16_t>::max)();
312  }
313 
314  // multithreading locking
315  LOCK(MetaChain::getInstance().getStorageManager()->csSubChainManager);
316 
317  // adding subchain into our map
318  SubChain *tmp = m_mapSCFactories[((MCP03::MetaChain::createSubchain*)block->pTransaction->txIn.pPayload)->caSubChainClassName]();
319  uint16_t uint16ChainIdentifier = tmp->init(
320  ((MCP03::MetaChain::createSubchain*)block->pTransaction->txIn.pPayload)->caChainName,
321  ((MCP03::MetaChain::createSubchain*)block->pTransaction->txIn.pPayload)->caSubChainClassName,
322  ((MCP03::MetaChain::createSubchain*)block->pTransaction->txIn.pPayload)->caPoP,
323  ((MCP03::MetaChain::createSubchain*)block->pTransaction->txIn.pPayload)->mapParams);
324 
325  if (uint16ChainIdentifier == (std::numeric_limits<uint16_t>::max)())
326  {
327  LOG_ERROR("Couldn't initialize SC, aborting.", "SCM");
328  return (std::numeric_limits<uint16_t>::max)();
329  }
330  m_mapSubChains[uint16ChainIdentifier] = tmp;
331 
332  // store this block into our MC
333  std::stringstream stream(std::ios_base::in | std::ios_base::out | std::ios_base::binary);
334  {
335  boost::archive::binary_oarchive oa(stream, boost::archive::no_header | boost::archive::no_tracking);
336  oa << block;
337  }
338  MetaChain::getInstance().getStorageManager()->writeRaw(MC_CHAIN_IDENTIFIER, stream.tellp(), (void*)stream.str().data());
339 
340  return uint16ChainIdentifier;
341  }
