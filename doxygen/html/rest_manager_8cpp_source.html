70 void restManager::handleRequest(web::http::http_request req)
71 {
72 #ifdef _DEBUG
73  LOG_DEBUG(req.to_string(), "REST");
74 #endif
75 
76  // we split up the calling path into a vector for easier processing
77  auto vecPaths = web::http::uri::split_path(web::http::uri::decode(req.relative_uri().path()));
78 
79  // check if we have atleast one parameter to work with
80  if (vecPaths.size() < 1)
81  {
82  req.reply(web::http::status_codes::BadRequest, U("Missing Request URI"));
83  return;
84  }
85 
86  // handle GET calls
87  if (req.method() == U("GET"))
88  {
89  // processing the request
90  if (vecPaths[0] == U("version"))
91  processVersion(&req);
92  else if (vecPaths[0] == U("time"))
93  processTime(&req);
94  else if (vecPaths[0] == U("info"))
95  processInfo(&req);
96  else
97  req.reply(web::http::status_codes::NotFound, U("API GET Method not found"));
98  }
99  // handle POST calls
100  else if (req.method() == U("POST"))
101  {
102  if (vecPaths[0] == U("test"))
103  processPostExample(req);
104  else
105  req.reply(web::http::status_codes::NotFound, U("API POST Method not found"));
106  }
107  else
108  req.reply(web::http::status_codes::BadRequest, U("Not supported http_request type"));
109 }
