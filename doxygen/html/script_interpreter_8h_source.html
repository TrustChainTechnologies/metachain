34  // Passing a non-strict-DER signature or one with undefined hashtype to a checksig operation causes script failure.
35  // Evaluating a pubkey that is not (0x04 + 64 bytes) or (0x02 or 0x03 + 32 bytes) by checksig causes script failure.
36  // (softfork safe, but not used or intended as a consensus rule).
37  STRICTENC = (1U << 1),

43  // (softfork safe, BIP62 rule 5).
44  LOW_S = (1U << 3),

46  // verify dummy stack item consumed by CHECKMULTISIG is of zero-length (softfork safe, BIP62 rule 7).
47  NULLDUMMY = (1U << 4),

49  // Using a non-push operator in the scriptSig causes script failure (softfork safe, BIP62 rule 2).
50  SIGPUSHONLY = (1U << 5),
51 
52  // Require minimal encodings for all push operations (OP_0... OP_16, OP_1NEGATE where possible, direct
53  // pushes up to 75 bytes, OP_PUSHDATA up to 255 bytes, OP_PUSHDATA2 for anything larger). Evaluating
54  // any other push causes the script to fail (BIP62 rule 3).
55  // In addition, whenever a stack element is interpreted as a number, it must be of minimal length (BIP62 rule 4).
56  // (softfork safe)
57  MINIMALDATA = (1U << 6),
58 
59  // Discourage use of NOPs reserved for upgrades (NOP1-10)
60  //
61  // Provided so that nodes can avoid accepting or mining transactions
62  // containing executed NOP's whose meaning may change after a soft-fork,
63  // thus rendering the script invalid; with this flag set executing
64  // discouraged NOPs fails the script. This verification flag will never be
65  // a mandatory flag applied to scripts in a block. NOPs that are not
66  // executed, e.g. within an unexecuted IF ENDIF block, are *not* rejected.
67  DISCOURAGE_UPGRADABLE_NOPS = (1U << 7),
68 
69  // Require that only a single stack element remains after evaluation. This changes the success criterion from
70  // "At least one stack element must remain, and when interpreted as a boolean, it must be true" to
71  // "Exactly one stack element must remain, and when interpreted as a boolean, it must be true".
72  // (softfork safe, BIP62 rule 6)
73  // Note: CLEANSTACK should never be used without P2SH or WITNESS.
74  CLEANSTACK = (1U << 8),
