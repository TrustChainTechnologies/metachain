58  bool Account::calcPubKey(ECDSA type)
59  {
60  // sanitize pub key
61  memset(m_keyPub, 0x00, 64 * sizeof(uint8_t));
62 
63  // check whether priv key is set
64  if (memcmp(m_keyPriv, m_keyPub, 64 * sizeof(uint8_t)) == 0)
65  return false;
66 
67  if (type == ECDSA::SECP256k1)
68  {
69  secp256k1_context *ctx = secp256k1_context_create(SECP256K1_CONTEXT_SIGN | SECP256K1_CONTEXT_VERIFY);
70  if ((secp256k1_ec_seckey_verify(ctx, (unsigned char*)getPrivKeyStr().c_str()) == 1) && (secp256k1_ec_pubkey_create(ctx, (secp256k1_pubkey*)m_keyPub, (unsigned char*)getPrivKeyStr().c_str()) == 1))
71  {
72  m_ecdsaPubKey = ECDSA::SECP256k1;
73  secp256k1_context_destroy(ctx);
74  return true;
75  }
76  else
77  {
78  m_ecdsaPubKey = ECDSA::not_calculated;
79  secp256k1_context_destroy(ctx);

97  // for prettier (std::string) output use getWalletAddress
98  bool Account::calcWalletAddress(uint8_t* walletAddress, uint16_t uiChainIdentifier, bool bTestNet)
99  {
100  /*
101  v1:
102  Format: [4byte checksum][1byte flags][64byte hashed pubkey(/w chain identifier)][1byte address version]
103  Raw Size: 70 byte
104 
105  Function flow:
106  X = SHA3-512([pubkey][2byte chain identifier])
107  C = SHA3-KECCAK-128(SHA3-KECCAK-128([flag byte]X[version byte]))
108  WA = [C,first 4 bytes][flag byte][X][version byte]
109  WA_nice = MC-<Chain Identifier String, up to 4 char>-<formatted WA>
110  */
111 
112  // check whether we have a pub key stored
113  static const uint8_t zero[64] = { 0 };
114  if (memcmp(m_keyPub, zero, 64 * sizeof(uint8_t)) == 0)
115  return false;
116 
117  // right now we only support SECP256k1 - change this security check to also include r1
118  if (m_ecdsaPubKey != ECDSA::SECP256k1)
119  return false;
120 
121  // all security checks gone through, start calculating the address
122  SHA3 crypto;
123 
124  // pub key with padded chain identifier, then calculate SHA3-512
125  uint8_t buffer[66] = { 0 };
126  memcpy(buffer, m_keyPub, 64);
127  memcpy(buffer + 64, &uiChainIdentifier, 2);
128  uint8_t* X = crypto.hash(SHA3::HashType::DEFAULT, SHA3::HashSize::SHA3_512, buffer, 65);
129 
130  // assemble byte flag
131  uint8_t byteFlag = bTestNet ? BITFIELD_TESTNET : 0x00; // fifth bit indicates testnet
132  byteFlag |= m_ecdsaPubKey == ECDSA::SECP256k1 ? BITFIELD_ECDSA_k1 : BITFIELD_ECDSA_r1; // first bit indicates ECDSA algorithm. 1=k1, 0=r1. may be changed in future versions to use more bit
133 
134  // calc hashsum
135  buffer[0] = byteFlag;
136  memcpy(buffer + 1, X, 64);
137  buffer[65] = WALLET_ADDRESS_VERSION;
138  uint8_t* C = crypto.hash(SHA3::HashType::KECCAK, SHA3::HashSize::SHA3_128, crypto.hash(SHA3::HashType::KECCAK, SHA3::HashSize::SHA3_128, buffer, 66), 32);
139 
140  // assemble the result
141  memcpy(walletAddress, C, 4);
142  memcpy(walletAddress + 4, buffer, 66);
143 
144  return true;
145  }

196  bool Account::verifyWalletAddress(std::string strWalletAddress)
197  {
198  /*
199  Steps to check for the validity:
200  1) remove MC- from the beginning
201  2) extract the Chain Identifier from the beginning and get their chainidentifier ID
202  3) get the version of this address (last byte). steps below may change depending on version number
203  4) extract the checksum
204  5) extract the flag byte
205  6) calculate checksum and compare it to the checksum provided
206  */
207 
208  std::string strTmp = strWalletAddress;
209 
210  // 70 chars is the adress alone without tokens or prefixes
211  if (strWalletAddress.length() < 70)
212  return false;
213 
214  // 1
215  if (!boost::istarts_with(strWalletAddress, WALLET_ADDRESS_STD_PREFIX+(std::string)"-"))
216  return false;
217  strTmp = strWalletAddress.substr(3);
218 
219  // 2
220  // calculating the chain identifier uid
221  std::string strChainIdentifier = strTmp.substr(0, strTmp.find("-"));
222 #ifdef _UTILS
223  // hardcoded in metachain-utils/keygenerator
224  if (strChainIdentifier == "MC")
225  m_uint16ChainIdentifier = 0;
226  else if (strChainIdentifier == "TCT")
227  m_uint16ChainIdentifier = 1;
228  else if (strChainIdentifier == "MINE")
229  m_uint16ChainIdentifier = 2;
230  else
231  return false;
232 #else
233  m_uint16ChainIdentifier = MetaChain::getInstance().getStorageManager()->getChainIdentifier(strChainIdentifier);
234 #endif
235 
236  strTmp = strTmp.substr(strChainIdentifier.length() + 1);
237 
238  // 3
239  uint8_t uiAddress[70];
240  if (!base58::decode(untokenize(strTmp), uiAddress))
241  return false;
242  m_uint8VersionNumber = uiAddress[69];
243 
244  // 4
245  uint8_t uiChecksum[4];
246  memcpy(uiChecksum, uiAddress, 4);
247 
248  // 5
249  m_uiFlags = uiAddress[4];
250 
251  // 6
252  SHA3 crypto;
253  uint8_t* C = crypto.hash(SHA3::HashType::KECCAK, SHA3::HashSize::SHA3_128, crypto.hash(SHA3::HashType::KECCAK, SHA3::HashSize::SHA3_128, (uiAddress + 4), 66), 32);
254  if (memcmp(uiChecksum, C, 4) != 0)
255  return false;
256 
257  // if we reached this point, everything went smooth and it's valid
258  return true;
259  }
