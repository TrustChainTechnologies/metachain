101  // This function calculates the wallet address based on testnet flag, ECDSA signing algorithm and chain identifier
102  // for prettier (std::string) output use getWalletAddress
103  bool Account::calcWalletAddress(uint8_t* walletAddress, uint16_t uiChainIdentifier, bool bTestNet)
104  {
105  /*
106  v1:
107  Format: [4byte checksum][1byte flags][64byte hashed pubkey(/w chain identifier)][1byte address version]
108  Raw Size: 70 byte
109 
110  Function flow:
111  X = SHA3-512([pubkey][2byte chain identifier])
112  C = SHA3-KECCAK-128(SHA3-KECCAK-128([flag byte]X[version byte]))
113  WA = [C,first 4 bytes][flag byte][X][version byte]
114  WA_nice = MC-<Chain Identifier String, up to 4 char>-<formatted WA>
115  */
116 
117  // check whether we have a pub key stored
118  static const uint8_t zero[64] = { 0 };
119  if (memcmp(m_keyPub, zero, 64 * sizeof(uint8_t)) == 0)
120  return false;
121 
122  // right now we only support SECP256k1 - change this security check to also include r1
123  if (m_ecdsaPubKey != ECDSA::SECP256k1)
124  return false;
125 
126  // all security checks gone through, start calculating the address
127  SHA3 crypto;
128 
129  // pub key with padded chain identifier, then calculate SHA3-512
130  uint8_t buffer[66] = { 0 };
131  memcpy(buffer, m_keyPub, 64);
132  memcpy(buffer + 64, &uiChainIdentifier, 2);
133  uint8_t* X = crypto.hash(SHA3::HashType::DEFAULT, SHA3::HashSize::SHA3_512, buffer, 65);
134 
135  // assemble byte flag
136  uint8_t byteFlag = bTestNet ? BITFIELD_TESTNET : 0x00; // fifth bit indicates testnet
137  byteFlag |= m_ecdsaPubKey == ECDSA::SECP256k1 ? BITFIELD_ECDSA_k1 : BITFIELD_ECDSA_r1; // first bit indicates ECDSA algorithm. 1=k1, 0=r1. may be changed in future versions to use more bit
138 
139  // calc hashsum
140  buffer[0] = byteFlag;
141  memcpy(buffer + 1, X, 64);
142  buffer[65] = WALLET_ADDRESS_VERSION;
143  uint8_t* C = crypto.hash(SHA3::HashType::KECCAK, SHA3::HashSize::SHA3_128, crypto.hash(SHA3::HashType::KECCAK, SHA3::HashSize::SHA3_128, buffer, 66), 32);
144 
145  // assemble the result
146  memcpy(walletAddress, C, 4);
147  memcpy(walletAddress + 4, buffer, 66);
148 
149  return true;
150  }

201  bool Account::verifyWalletAddress(std::string strWalletAddress)
202  {
203  /*
204  Steps to check for the validity:
205  1) remove MC- from the beginning
206  2) extract the Chain Identifier from the beginning and get their chainidentifier ID
207  3) get the version of this address (last byte). steps below may change depending on version number
208  4) extract the checksum
209  5) extract the flag byte
210  6) calculate checksum and compare it to the checksum provided
211  */
212 
213  std::string strTmp = strWalletAddress;
214 
215  // 70 chars is the adress alone without tokens or prefixes
216  if (strWalletAddress.length() < 70)
217  return false;
218 
219  // 1
220  if (!boost::istarts_with(strWalletAddress, WALLET_ADDRESS_STD_PREFIX+(std::string)"-"))
221  return false;
222  strTmp = strWalletAddress.substr(3);
223 
224  // 2
225  // calculating the chain identifier uid
226  std::string strChainIdentifier = strTmp.substr(0, strTmp.find("-"));
227 #ifdef _UTILS
228  // hardcoded in metachain-utils/keygenerator
229  if (strChainIdentifier == "MC")
230  m_uint16ChainIdentifier = 0;
231  else if (strChainIdentifier == "TCT")
232  m_uint16ChainIdentifier = 1;
233  else if (strChainIdentifier == "MINE")
234  m_uint16ChainIdentifier = 2;
235  else
236  return false;
237 #else
238  m_uint16ChainIdentifier = MetaChain::getInstance().getStorageManager()->getChainIdentifier(strChainIdentifier);
239 #endif
240 
241  strTmp = strTmp.substr(strChainIdentifier.length() + 1);
242 
243  // 3
244  uint8_t uiAddress[70];
245  if (!base58::decode(untokenize(strTmp), uiAddress))
246  return false;
247  m_uint8VersionNumber = uiAddress[69];
248 
249  // 4
250  uint8_t uiChecksum[4];
251  memcpy(uiChecksum, uiAddress, 4);
252 
253  // 5
254  m_uiFlags = uiAddress[4];
255 
256  // 6
257  SHA3 crypto;
258  uint8_t* C = crypto.hash(SHA3::HashType::KECCAK, SHA3::HashSize::SHA3_128, crypto.hash(SHA3::HashType::KECCAK, SHA3::HashSize::SHA3_128, (uiAddress + 4), 66), 32);
259  if (memcmp(uiChecksum, C, 4) != 0)
260  return false;
261 
262  // if we reached this point, everything went smooth and it's valid
263  return true;
264  }

266  bool Account::verify(const uint256 &hash, const std::vector<unsigned char>& vchSig)
267  {
268  if (!verifyWalletAddress())
269  return false;
270 
271  secp256k1_pubkey pubkey;
272  secp256k1_ecdsa_signature sig;
273  secp256k1_context* secp256k1_context_verify = NULL;
274 
275  // todo: handle different pub key lengths (compressed / uncompressed)

277  return false;

279  if (vchSig.size() == 0) {
280  return false;

282  if (!ecdsa_signature_parse_der_lax(secp256k1_context_verify, &sig, &vchSig[0], vchSig.size())) {

301  int Account::ecdsa_signature_parse_der_lax(const secp256k1_context* ctx, secp256k1_ecdsa_signature* sig, const unsigned char *input, size_t inputlen)
302  {
303  size_t rpos, rlen, spos, slen;
304  size_t pos = 0;
305  size_t lenbyte;
306  unsigned char tmpsig[64] = { 0 };
307  int overflow = 0;
308 
309  /* Hack to initialize sig with a correctly-parsed but invalid signature. */
310  secp256k1_ecdsa_signature_parse_compact(ctx, sig, tmpsig);
311 
312  /* Sequence tag byte */
313  if (pos == inputlen || input[pos] != 0x30) {
314  return 0;
315  }
316  pos++;
317 
318  /* Sequence length bytes */
319  if (pos == inputlen) {
320  return 0;
321  }
322  lenbyte = input[pos++];
323  if (lenbyte & 0x80) {
324  lenbyte -= 0x80;
325  if (pos + lenbyte > inputlen) {
326  return 0;
327  }
328  pos += lenbyte;
329  }
330 
331  /* Integer tag byte for R */
332  if (pos == inputlen || input[pos] != 0x02) {
333  return 0;
334  }
335  pos++;
336 
337  /* Integer length for R */
338  if (pos == inputlen) {
339  return 0;
340  }
341  lenbyte = input[pos++];
342  if (lenbyte & 0x80) {
343  lenbyte -= 0x80;
344  if (pos + lenbyte > inputlen) {
345  return 0;
346  }
347  while (lenbyte > 0 && input[pos] == 0) {
348  pos++;
349  lenbyte--;
350  }
351  if (lenbyte >= sizeof(size_t)) {
352  return 0;
353  }
354  rlen = 0;
355  while (lenbyte > 0) {
356  rlen = (rlen << 8) + input[pos];
357  pos++;
358  lenbyte--;
359  }
360  }
361  else {
362  rlen = lenbyte;
363  }
364  if (rlen > inputlen - pos) {
365  return 0;
366  }
367  rpos = pos;
368  pos += rlen;
369 
370  /* Integer tag byte for S */
371  if (pos == inputlen || input[pos] != 0x02) {
372  return 0;
373  }
374  pos++;
375 
376  /* Integer length for S */
377  if (pos == inputlen) {
378  return 0;
379  }
380  lenbyte = input[pos++];
381  if (lenbyte & 0x80) {
382  lenbyte -= 0x80;
383  if (pos + lenbyte > inputlen) {
384  return 0;
385  }
386  while (lenbyte > 0 && input[pos] == 0) {
387  pos++;
388  lenbyte--;
389  }
390  if (lenbyte >= sizeof(size_t)) {
391  return 0;
392  }
393  slen = 0;
394  while (lenbyte > 0) {
395  slen = (slen << 8) + input[pos];
396  pos++;
397  lenbyte--;
398  }
399  }
400  else {
401  slen = lenbyte;
402  }
403  if (slen > inputlen - pos) {
404  return 0;
405  }
406  spos = pos;
407  pos += slen;
408 
409  /* Ignore leading zeroes in R */
410  while (rlen > 0 && input[rpos] == 0) {
411  rlen--;
412  rpos++;
413  }
414  /* Copy R value */
415  if (rlen > 32) {
416  overflow = 1;
417  }
418  else {
419  memcpy(tmpsig + 32 - rlen, input + rpos, rlen);
420  }
421 
422  /* Ignore leading zeroes in S */
423  while (slen > 0 && input[spos] == 0) {
424  slen--;
425  spos++;
426  }
427  /* Copy S value */
428  if (slen > 32) {
429  overflow = 1;
430  }
431  else {
432  memcpy(tmpsig + 64 - slen, input + spos, slen);
433  }
434 
435  if (!overflow) {
436  overflow = !secp256k1_ecdsa_signature_parse_compact(ctx, sig, tmpsig);
437  }
438  if (overflow) {
439  /* Overwrite the result again with a correctly-parsed but invalid
440  signature if parsing failed. */
441  memset(tmpsig, 0, 64);
442  secp256k1_ecdsa_signature_parse_compact(ctx, sig, tmpsig);
443  }
444  return 1;
445  }
