244  void Xor(const std::vector<unsigned char>& key)
245  {
246  if (key.size() == 0) {
247  return;
248  }
249 
250  for (size_type i = 0, j = 0; i != size(); i++) {
251  vch[i] ^= key[j++];
252 
253  // This potentially acts on very many bytes of data, so it's
254  // important that we calculate `j`, i.e. the `key` index in this
255  // way instead of doing a %, which would effectively be a division
256  // for each byte Xor'd -- much slower than need be.
257  if (j == key.size())
258  j = 0;
259  }
