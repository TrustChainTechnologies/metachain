72  void crBlock::calcMerkleRoot()
73  {
74  SHA3 crypto;
75  std::vector< uint256 > leaves, tmp;
76 
77  // get all our leaves in place
78  for (auto it : vecTx)
79  {
80  it->hash = it->calcHash();
81  leaves.push_back(it->hash);
82  }
83 
84  // if we don't have leaves we reset the merkle root
85  if (leaves.size() == 0)
86  {
87  hashMerkleRoot.SetNull();
88  return;
89  }
90 
91  // if we have only one tx, we add a second one so that we have atleast one iteration in the merkle root calculation
92  // if we have more than one but an odd number, also add the last one
93  if ((leaves.size() == 1) || (leaves.size() > 1 && !(leaves.size() % 2)))
94  leaves.push_back(leaves.back());
95 
96  // we continue the hash calculation until we only have one last result - the merkle tree
97  while (leaves.size() != 1)
98  {
99  // clean up our tmp vector
100  tmp.clear();
101 
102  while (!leaves.empty())
103  {
104  // combine both leaves into one
105  uint8_t cmb[64];
106  memcpy(&cmb[0], leaves.back().begin(), 32);
107  leaves.pop_back();
108  memcpy(&cmb[31], leaves.back().begin(), 32);
109  leaves.pop_back();
110 
111  // calc combined hash and add it
112  tmp.push_back(crypto.hash256(SHA3::HashType::DEFAULT, cmb, 64));
113  }
114 
115  // if we have more than 1 result in the vector and an odd number, we add the last entry again for hashing
116  if (tmp.size() != 1 && !(tmp.size() % 2))
117  tmp.push_back(tmp.back());
118 
119  // move our tmp data to our leaves vector
120  leaves.swap(tmp);
121  }
122 
123  // update the merkle root
124  hashMerkleRoot = leaves[0];
125  }
126 }
