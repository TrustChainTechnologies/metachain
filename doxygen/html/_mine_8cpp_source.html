45 
46  const char* pszTimestamp = "The Times 03/Jan/2009 Chancellor on brink of second bailout for banks";
47  std::shared_ptr<MCP03::crTransaction> txNew((MCP03::crTransaction*)m_pPoP->createTXElement());
48  txNew->uint16tVersion = 1;

51  txNew->vecIn[0].scriptSignature = MCP06::CScript() << 486604799 << MCP06::CScriptNum(4) << std::vector<unsigned char>((const unsigned char*)pszTimestamp, (const unsigned char*)pszTimestamp + strlen(pszTimestamp));
52  txNew->vecIn[0].txPrev = std::shared_ptr< MCP03::txOutRef >(new MCP03::txOutRef() );

90  bool Mine::processBlock(MCP03::Block* Block)
91  {
92  // security checks
93  if (!checkBlock(Block))
94  return false;
95 
96  // security check for each tx
97  for (auto &it : ((MCP03::crBlock*)Block)->vecTx)
98  {
99  // check consistency of tx hash
100  if (!it->checkHash())
101  {
102  LOG_ERROR("The calculated hash and the provided hash in this tx is not consistent. Dismissing Block!", "MINE");
103  return false;
104  }
105  }
106 
107  // we're using a batch statment for the block. this allows safety for the data in case of interruptions. the incoming TX will not get flagged as spent without the outgoing TX being generated
108  LOCK(m_pDB->batchCriticalSection);
109  m_pDB->batchStart();
110 
111  // variable that tracks if we have only one coinbase tx in this block
112  bool bCoinbase = false;
113 
114  // a list of txin outrefs to check double spending within a single block
115  std::unordered_set< std::string > usetSpendings;
116 
117  // go through each tx and process it
118  for (auto &it : ((MCP03::crBlock*)Block)->vecTx)
119  {
120  // todo: differentiate different types of transactions. right now hardcoded: only currency transactions
121  // workflow:
122  // 1: check incoming if valid to spend and exist.
123  // 2: check if incoming val >= outgoing val
124  // 3: remove incoming from spend list
125  // 4: add outgoing to spend list
126  // 5: update MCP01::Account array of spenable tx
127 
128  uint64_t uiValIncoming = 0, uiValOutgoing = 0;
129  unsigned int uiPosition = 0;
130 
131  // when it's a coinbase tx, we don't check for incoming tx or mark the incoming as spent, we only check if the value of the coinbase tx is correct.
132  // we also only allow one coinbase tx per block, otherwise we dismiss the block
133  if (!it->isCoinBase())
134  {
135  // step1
136  // todo: check if spendable with script signature
137  for (auto &incoming : it->vecIn)
138  {
139  std::string strIdent = "txO." + incoming.txPrev.get()->hash.GetHex() + "." + std::to_string(uiPosition);
140 
141  // check wether this ident was already spent in this block. If so, it's a double spending and we dismiss this TX
142  if (usetSpendings.count(strIdent) == 1)
143  {
144  LOG_ERROR("Double spending in this block detected. Dismissing TX!", "MINE");
145  goto next;
146  }
147  else
148  usetSpendings.emplace(strIdent);
149 
150  if (m_pDB->get(strIdent + ".spent", true))
151  {
152  // the incoming TX was either not found or it was already spent. in any way, we can't accept this as incoming tx
153  LOG_ERROR("Incoming TX in this TX was either not found or already spent. Dismissing TX!", "MINE");
154  goto next;
155  }
156  uiValIncoming += m_pDB->get(strIdent + ".val", (unsigned int)0);
157 
158  uiPosition++;
159  }
160 
161  // step2
162  for (auto &outgoing : it->vecOut)
163  uiValOutgoing += outgoing.uint64tValue;
164 
165  if (uiValOutgoing > uiValIncoming)
166  {
167  LOG_ERROR("Outgoing amount is bigger than incoming TXs. Dismissing TX!", "MINE");
168  goto next;
169  }
170 
171  // step3
172  uiPosition = 0;
173  for (auto &incoming : it->vecIn)
174  {
175  // we delete the entry from the DB. Since we're retrieving it with a default of "spent" = true, it doesn't matter if we update it to be "spent"=true or delete the entry. so we delete it to save storage space
176  m_pDB->batchDeleteEntry("txO." + incoming.txPrev.get()->hash.GetHex() + "." + std::to_string(uiPosition) + ".spent");
177  uiPosition++;
178  }
179  }
180  else
181  {
182  if (bCoinbase)
183  {
184  LOG_ERROR("The block includes two coinbase tx. Dismissing Block!", "MINE");
185  return false;
186  }
187  // todo: check coinbase value
188  bCoinbase = true;
189  }
190 
191  // step4
192  uiPosition = 0;
193  for (auto &outgoing : it->vecOut)
194  {
195  std::string strIdent = "txO." + it->hash.GetHex() + "." + std::to_string(uiPosition);
196  m_pDB->batchAddStatement(strIdent + ".spent", "0");
197  m_pDB->batchAddStatement(strIdent + ".val", std::to_string(outgoing.uint64tValue));
198  uiPosition++;
199  }
200 
201  // step5
202  {
203  std::unordered_map< std::string, MCP01::Account > umapAccounts;
204 
205  // remove incoming TX from spendable list, except when it's a coinbase
206  if( !it->isCoinBase() )
207  {
208  for (auto &incoming : it->vecIn)
209  {
210  // check if we have the account already in our list, if not add it
211  if (umapAccounts.count(incoming.strSignature) != 1)
212  {
213  std::string strTmp = m_pDB->get("txU." + incoming.strSignature, (std::string)"");
214  if (strTmp != "")
215  {
216  std::stringstream stream(strTmp);
217  boost::archive::binary_iarchive ia(stream, boost::archive::no_header | boost::archive::no_tracking);
218  MCP01::Account tmp;
219  ia >> tmp;
220  umapAccounts[incoming.strSignature] = tmp;
221  }
222  else
223  umapAccounts[incoming.strSignature] = MCP01::Account();
224  }
225 
226  // since it's an incoming TX, we have to remove the entry from the spendable list
227  umapAccounts[incoming.strSignature].umapUnspentTX.erase(incoming.txPrev.get()->hash.GetHex());
228  }
229  }
230 
231  // add outgoing TX to spenable list
232  uiPosition = 0;
233  for (auto &outgoing : it->vecOut)
234  {
235  std::string strPubKey = MCP01::base58::encode(outgoing.uint8tPubKey, 64);
236 
237  // check if we have the account already in our list, if not add it
238  if (umapAccounts.count(strPubKey) != 1)
239  {
240  std::string strTmp = m_pDB->get("txU." + strPubKey, (std::string)"");
241  if (strTmp != "")
242  {
243  std::stringstream stream(strTmp);
244  boost::archive::binary_iarchive ia(stream, boost::archive::no_header | boost::archive::no_tracking);
245  MCP01::Account tmp;
246  ia >> tmp;
247  umapAccounts[strPubKey] = tmp;
248  }
249  else
250  umapAccounts[strPubKey] = MCP01::Account();
251  }
252 
253  // since it's an outgoing TX, we have to add the entry to the spendable list
254  umapAccounts[strPubKey].umapUnspentTX.emplace(it->hash.GetHex(), uiPosition);
255 
256  uiPosition++;
257  }
258 
259  // since we now updated all spenable lists, we sync them back into our db
260  for (auto &account : umapAccounts)
261  {
262  std::stringstream stream(std::ios_base::in | std::ios_base::out | std::ios_base::binary);
263  boost::archive::binary_oarchive oa(stream, boost::archive::no_header | boost::archive::no_tracking);
264  oa << account.second;
265  m_pDB->batchAddStatement("txU." + account.first, rocksdb::Slice(stream.str().data(), stream.tellp()).ToString());
266  }
267  }
268 
269  // goto label used for ignoring a certain transaction
270  next:;
271  }
272 
273  // now we can execute our batch statement
274  m_pDB->batchFinalize();
275 
276  // store the block with our storagemanager
277  saveBlock(Block);
278 
279  // block successfully processed
280  return true;
281  }
