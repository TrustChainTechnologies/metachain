48 bool MetaChain::initialize(CSimpleIniA* iniFile, boost::filesystem::path pathExecutable)
49 {
50  m_pathExecutable = pathExecutable;
51 
52  // read autoupdate settings
53  m_iVersionTicksTillUpdate = iniFile->GetLongValue("autoupdate", "ticks_until_update_triggered", 10);
54  m_bAutoUpdate = iniFile->GetBoolValue("autoupdate", "enable", true);
55  m_strCDNUrl = iniFile->GetValue("autoupdate", "cdn_url", "https://cdn.tct.io/");
56  m_pathTmp = iniFile->GetValue("autoupdate", "tmp_dir", "tmp");
57  if (m_pathTmp.is_relative())
58  m_pathTmp = boost::filesystem::current_path() / iniFile->GetValue("autoupdate", "tmp_dir", "tmp");
59 
60  // do a autoupdate call upon start if requested
61  if (iniFile->GetBoolValue("autoupdate", "enable", true) && iniFile->GetBoolValue("autoupdate", "autoupdate_on_start", true))
62  {
63  // if the doAutoUpdate() function returns true, a update was successfully made.
64  // we skip the rest of the initialization function since we need to start the new version
65  if (doAutoUpdate())
66  return false;
67  }
68 
69  // getting our mode, since we only have FN and CL, we use a flag
70  m_bModeFN = ((std::string)iniFile->GetValue("general", "mode", "fn") == "fn" ? true : false);
71  LOG("initializing this node to run in mode: " + (std::string)(m_bModeFN ? "FN" : "CL"), "MC");
72 
73  // check wether this instance is testnet or not
74  m_bTestNet = iniFile->GetBoolValue("general", "testnet", false);
75 
76  // get the POPs that shall be loaded
77  boost::split(vecPOPs, (std::string)iniFile->GetValue("subchains", "pop_load", "*"), boost::is_any_of(","));
78 
79  // get the SC white and blacklist
80  boost::split(vecSC_Whitelist, (std::string)iniFile->GetValue("subchains", "subchain_whitelist", "*"), boost::is_any_of(","), boost::algorithm::token_compress_on);
81  boost::split(vecSC_Blacklist, (std::string)iniFile->GetValue("subchains", "subchain_blacklist", "*"), boost::is_any_of(","), boost::algorithm::token_compress_on);
82 
83  // sanitizing white and blacklist
84  for (auto &it : vecSC_Whitelist)
85  boost::trim(it);
86  for (auto &it : vecSC_Blacklist)
87  boost::trim(it);
88 
89  // create the block storage backends, check their integrity and check if no other instance is running
90  m_pStorageManager = new StorageManager(this);
91  if (!m_pStorageManager->initialize(iniFile))
92  return false;
93 
94  // start the lightweight scheduling thread
95  CScheduler::Function serviceLoop = boost::bind(&CScheduler::serviceQueue, &m_scheduler);
96  m_threadGroup.create_thread(boost::bind(&TraceThread<CScheduler::Function>, "scheduler", serviceLoop));
97 
98  // get the connected wallet address for this node
99  m_mcpWallet.setWalletAddress( iniFile->GetValue("general", "wallet") );
100  if (m_mcpWallet.verifyWalletAddress())
101  {
102  // check if testnet flag from ini and from wallet address match
103  if (m_bTestNet != m_mcpWallet.isTestNet())
104  {
105  m_mcpWallet = MCP01::Account();
106  LOG_ERROR("Wallet address doesn't match TestNet configuration, removing wallet address", "MC");
107  }
108  else if (!m_mcpWallet.isBaseAddress())
109  {
110  m_mcpWallet = MCP01::Account();
111  LOG_ERROR("Wallet address is not a base address.", "MC");
112  }
113  else
114  LOG("Wallet address " + m_mcpWallet.getWalletAddress() + " successfully connected to node", "MC");
115  }
116  else
117  {
118  m_mcpWallet = MCP01::Account();
119  LOG_ERROR("No valid wallet address connected to node", "MC");
120  }
121 
122  // create a new network manager
123  m_pNetworkManager = new NetworkManager(this);
124 
125  // initializing the network manager
126  if (!m_pNetworkManager->initialize(iniFile))
127  return false;
128 
129  // initialize a rest API if configured
130  if (iniFile->GetBoolValue("rest", "enable", true))
131  {
132  m_pRestManager = new restManager( CService(iniFile->GetValue("rest", "ip", "127.0.0.1"), iniFile->GetLongValue("rest", "port", 10016)), iniFile->GetBoolValue("rest", "enable_ssl", false) );
133  m_pRestManager->init();
134  }
135 
136  // finally everything is initialized and we print our copyright info
137  LicenseInfo();
138 
139  // everything is fine
140  return true;
141 }

154 {
155  m_usNewerVersionTicker++;
156 
157  // after X ticks, we read the ini value if we auto update. if not, we close the node. if we have autoupdate configured, we connect to a CDN and download the latest binary and try to restart the node smoothly
158  if (m_usNewerVersionTicker >= m_iVersionTicksTillUpdate)
159  {
160  if (!m_bAutoUpdate)
161  {
162  // if we don't want to auto update, we close the node due to incompability

164  }

176 // auto update function that returns false if nothing was updated and true if something was updated
177 // when true is returned, the node needs to restart in order for changes to take affect
178 bool MetaChain::doAutoUpdate()
179 {
180  if (m_strCDNUrl == "")
181  return false;
182 
183  LOG("Checking for newer Version at: " + m_strCDNUrl, "AU");
184 
185  // first fetch the current version number from the CDN
186  CURL *curl = curl_easy_init();
187  if (curl)
188  {
189  std::string strCurrentVersion;
190  curl_easy_setopt(curl, CURLOPT_URL, (m_strCDNUrl + "/" + (isTestNet() ? AU_VERSION_FILE_TESTNET : AU_VERSION_FILE)).c_str());
191  curl_easy_setopt(curl, CURLOPT_SSL_VERIFYPEER, 0);
192  curl_easy_setopt(curl, CURLOPT_WRITEFUNCTION, curlWriteCallback);
193  curl_easy_setopt(curl, CURLOPT_WRITEDATA, &strCurrentVersion);
194  CURLcode res = curl_easy_perform(curl);
195  curl_easy_cleanup(curl);
196 
197  if (res != CURLE_OK)
198  {
199  LOG_ERROR("Error receiving version number: " + (std::string)curl_easy_strerror(res), "AU");
200  return false;
201  }
202 
203  // remove tags that might be around and then convert to a number for version comparing
204  boost::trim_if(strCurrentVersion, boost::is_any_of("\n\r "));
205 
206  // convert to number
207  uint32_t currentVersion = 0;
208  try
209  {
210  currentVersion = boost::lexical_cast<uint32_t>(strCurrentVersion);
211  }
212  catch (const boost::bad_lexical_cast &e)
213  {
214  LOG_ERROR("Can't convert version number to comparable format: (" + strCurrentVersion + ") " + (std::string)e.what(), "AU");
215  return false;
216  }
217 
218  // compare version numbers
219  if (currentVersion != g_cuint32tVersion)
220  {
221  // get the newest version
222  LOG("New version found, updating", "AU");
223 
224  // assembling the download uri
225  std::string strURI = m_strCDNUrl + "/";
226 #ifdef _WIN32
227  strURI += "windows/";
228 #else
229  strURI += "linux/";
230 #endif
231 #ifdef _X86_64
232  strURI += "x86_64/";
233 #else
234  strURI += "i386/";
235 #endif
236  strURI += strCurrentVersion + "/";
237 
238  // make sure we have a clean tmp directory
239  try
240  {
241  boost::filesystem::remove_all(m_pathTmp);
242  boost::filesystem::create_directory(m_pathTmp);
243  }
244  catch (const boost::filesystem::filesystem_error& e)
245  {
246  LOG_ERROR("Couldn't clean tmp directory for safe processing: " + (std::string)e.what(), "AU");
247  return false;
248  }
249 
250  // download the files
251  if (!downloadFile(strURI + "check.sum", (m_pathTmp / AU_CHECKSUM_FILE).string()))
252  return false;
253  if (!downloadFile(strURI + "node.zip", (m_pathTmp / AU_NODE_FILE).string()))
254  return false;
255 
256  // calc and check the hashsum
257  LOG("Checking checksums", "AU");
258  SHA3 crypto;
259  std::ifstream ifs((m_pathTmp / AU_CHECKSUM_FILE).string());
260  std::string strChecksum((std::istreambuf_iterator<char>(ifs)), (std::istreambuf_iterator<char>()));
261  boost::trim_if(strChecksum, boost::is_any_of("\n\r "));
262  ifs.close();
263  if (crypto.to_string(crypto.hashFile((m_pathTmp / AU_NODE_FILE).string(), SHA3::HashType::DEFAULT, SHA3::HashSize::SHA3_512), SHA3::HashSize::SHA3_512) != strChecksum)
264  {
265  LOG_ERROR("Checksums don't match", "AU");
266  return false;
267  }
268  LOG("Checksums match, continuing updating", "AU");
269 
270  // extract content from the zip file
271  LOG("Extracting ZIP file", "AU");
272  ZIP::Unzip zipFile;
273  zipFile.open( (m_pathTmp / AU_NODE_FILE).string().c_str() );
274  auto filenames = zipFile.getFilenames();
275  for (auto it = filenames.begin(); it != filenames.end(); it++)
276  {
277  std::ofstream outFile((m_pathTmp / *it).string(), std::ofstream::binary);
278 
279  zipFile.openEntry((*it).c_str());
280  zipFile >> outFile;
281 
282  outFile.close();
283  }
284  zipFile.close();
285  boost::filesystem::remove(m_pathTmp / AU_NODE_FILE);
286  boost::filesystem::remove(m_pathTmp / AU_CHECKSUM_FILE);
287 
288  LOG("Replacing old files with new files (except configs)", "AU");
289  boost::filesystem::path pathRoot(m_pathExecutable);
290  pathRoot.remove_filename();
291 
292  // move the executable to a backup location
293  if (boost::filesystem::exists(pathRoot / AU_NODE_BACKUPFILE))
294  boost::filesystem::remove(pathRoot / AU_NODE_BACKUPFILE);
295  boost::filesystem::rename(m_pathExecutable, pathRoot / AU_NODE_BACKUPFILE);
296 
297  // copy all extracted files to our local path
298  for (boost::filesystem::directory_iterator file(m_pathTmp); file != boost::filesystem::directory_iterator(); file++)
299  {
300  try
301  {
302  boost::filesystem::rename(file->path(), pathRoot / file->path().filename());
303  }
304  catch (const boost::filesystem::filesystem_error& e)
305  {
306  LOG_ERROR("Couldn't move updated file sources: " + (std::string)e.what(), "AU");
307  return false;
308  }
309  }
310 
311  // the update process is now completed. the function that called this function needs to handle the shutdown and restart of the updated version
312  LOG("Autoupdate process completed. Restarting node to finish the updating process... ", "AU");
313  sabShutdown = true; // shutdown the thread that runs the MC
314  sabReboot = true; // reboot the MC Node
315 
316  return true;
317  }
318  else
319  {
320  LOG("No new version found, continuing work", "AU");
321  return false;
322  }
323  }
324  else
325  {
326  LOG_ERROR("Error initializing curl, AutoUpdate aborted", "AU");
327  return false;
328  }
329 }

348  curl_easy_cleanup(curl);
