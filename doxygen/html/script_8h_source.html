225  explicit CScriptNum(const std::vector<unsigned char>& vch, bool fRequireMinimal,
226  const size_t nMaxNumSize = nDefaultMaxNumSize)
227  {
228  if (vch.size() > nMaxNumSize) {
229  throw scriptnum_error("script number overflow");
230  }
231  if (fRequireMinimal && vch.size() > 0) {
232  // Check that the number is encoded with the minimum possible
233  // number of bytes.
234  //
235  // If the most-significant-byte - excluding the sign bit - is zero
236  // then we're not minimal. Note how this test also rejects the
237  // negative-zero encoding, 0x80.
238  if ((vch.back() & 0x7f) == 0) {
239  // One exception: if there's more than one byte and the most
240  // significant bit of the second-most-significant-byte is set
241  // it would conflict with the sign bit. An example of this case
242  // is +-255, which encode to 0xff00 and 0xff80 respectively.
243  // (big-endian).
244  if (vch.size() <= 1 || (vch[vch.size() - 2] & 0x80) == 0) {
245  throw scriptnum_error("non-minimally encoded script number");
246  }
247  }
248  }
249  m_value = set_vch(vch);
250  }

327  static std::vector<unsigned char> serialize(const int64_t& value)
328  {
329  if (value == 0)
330  return std::vector<unsigned char>();
331 
332  std::vector<unsigned char> result;
333  const bool neg = value < 0;
334  uint64_t absvalue = neg ? -value : value;
335 
336  while (absvalue)
337  {
338  result.push_back(absvalue & 0xff);
339  absvalue >>= 8;
340  }
341 
342  // - If the most significant byte is >= 0x80 and the value is positive, push a
343  // new zero-byte to make the significant byte < 0x80 again.
344 
345  // - If the most significant byte is >= 0x80 and the value is negative, push a
346  // new 0x80 byte that will be popped off when converting to an integral.
347 
348  // - If the most significant byte is < 0x80 and the value is negative, add
349  // 0x80 to it, since it will be subtracted and interpreted as a negative when
350  // converting to an integral.
351 
352  if (result.back() & 0x80)
353  result.push_back(neg ? 0x80 : 0);
354  else if (neg)
355  result.back() |= 0x80;
356 
357  return result;
358  }
359 
360  private:
361  static int64_t set_vch(const std::vector<unsigned char>& vch)
362  {
363  if (vch.empty())
364  return 0;
365 
366  int64_t result = 0;
367  for (size_t i = 0; i != vch.size(); ++i)
368  result |= static_cast<int64_t>(vch[i]) << 8 * i;
369 
370  // If the input vector's most significant byte is 0x80, remove it from
371  // the result's msb and return a negative.
372  if (vch.back() & 0x80)
373  return -((int64_t)(result & ~(0x80ULL << (8 * (vch.size() - 1)))));
374 
375  return result;
376  }

510  bool GetOp2(const_iterator& pc, opcodetype& opcodeRet, std::vector<unsigned char>* pvchRet) const
511  {
512  opcodeRet = OP_INVALIDOPCODE;
513  if (pvchRet)
514  pvchRet->clear();
515  if (pc >= end())
516  return false;
517 
518  // Read instruction
519  if (end() - pc < 1)
520  return false;
521  unsigned int opcode = *pc++;
522 
523  // Immediate operand
524  if (opcode <= OP_PUSHDATA4)
525  {
526  unsigned int nSize = 0;
527  if (opcode < OP_PUSHDATA1)
528  {
529  nSize = opcode;
530  }
531  else if (opcode == OP_PUSHDATA1)
532  {
533  if (end() - pc < 1)
534  return false;
535  nSize = *pc++;
536  }
537  else if (opcode == OP_PUSHDATA2)
538  {
539  if (end() - pc < 2)
540  return false;
541  nSize = ReadLE16(&pc[0]);
542  pc += 2;
543  }
544  else if (opcode == OP_PUSHDATA4)
545  {
546  if (end() - pc < 4)
547  return false;
548  nSize = ReadLE32(&pc[0]);
549  pc += 4;
550  }
551  if (end() - pc < 0 || (unsigned int)(end() - pc) < nSize)
552  return false;
553  if (pvchRet)
554  pvchRet->assign(pc, pc + nSize);
555  pc += nSize;
556  }
557 
558  opcodeRet = (opcodetype)opcode;
559  return true;
560  }

578  int FindAndDelete(const CScript& b)
579  {
580  int nFound = 0;
581  if (b.empty())
582  return nFound;
583  CScript result;
584  iterator pc = begin(), pc2 = begin();
585  opcodetype opcode;
586  do
587  {
588  result.insert(result.end(), pc2, pc);
589  while (static_cast<size_t>(end() - pc) >= b.size() && std::equal(b.begin(), b.end(), pc))
590  {
591  pc = pc + b.size();
592  ++nFound;
593  }
594  pc2 = pc;
595  } while (GetOp(pc, opcode));
596 
597  if (nFound > 0) {
598  result.insert(result.end(), pc2, end());
599  *this = result;
600  }
601 
602  return nFound;
603  }
