231 bool CNetAddr::IsValid() const
232 {
233  // Cleanup 3-byte shifted addresses caused by garbage in size field
234  // of addr messages from versions before 0.2.9 checksum.
235  // Two consecutive addr messages look like this:
236  // header20 vectorlen3 addr26 addr26 addr26 header20 vectorlen3 addr26 addr26 addr26...
237  // so if the first length field is garbled, it reads the second batch
238  // of addr misaligned by 3 bytes.
239  if (memcmp(ip, pchIPv4 + 3, sizeof(pchIPv4) - 3) == 0)
240  return false;
241 
242  // unspecified IPv6 address (::/128)
243  unsigned char ipNone6[16] = {};
244  if (memcmp(ip, ipNone6, 16) == 0)
245  return false;
246 
247  // documentation IPv6 address
248  if (IsRFC3849())
249  return false;
250 
251  if (IsIPv4())
252  {
253  // INADDR_NONE
254  uint32_t ipNone = INADDR_NONE;
255  if (memcmp(ip + 12, &ipNone, 4) == 0)
256  return false;
257 
258  // 0
259  ipNone = 0;
260  if (memcmp(ip + 12, &ipNone, 4) == 0)
261  return false;
262  }
263 
264  return true;
